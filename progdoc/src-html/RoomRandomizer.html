001import java.util.ArrayList;
002import java.util.HashMap;
003import java.util.Random;
004
005/**
006 * 
007 * 
008 * @author C.Diouy
009 * @version 2022.03.02
010 */
011public class RoomRandomizer {
012    private HashMap<String, Room> aAllRooms;
013    private ArrayList<Room> aRoomsArrayList;
014
015    /**
016     * this constructor create a room randomizer
017     * 
018     * @param pAllRooms hashmap contain all rooms
019     */
020    public RoomRandomizer(HashMap<String, Room> pAllRooms) {
021        this.aAllRooms = pAllRooms;
022        this.aRoomsArrayList = new ArrayList<Room>();
023        fillArrayList();
024    }
025
026    /**
027     * this method fill the array list
028     */
029    private void fillArrayList() {
030        HashMap<String, Room> vAllRooms = this.aAllRooms;
031        for (String vRoomName : vAllRooms.keySet()) {
032            this.aRoomsArrayList.add(vAllRooms.get(vRoomName));
033        }
034    }
035
036    /**
037     * Find a random room
038     * 
039     * @return room in array list
040     */
041    public Room findRandomRoom() {
042        Random vRandom = new Random();
043        int vRandomIntInArray = vRandom.nextInt(this.aRoomsArrayList.size());
044
045        return this.aRoomsArrayList.get(vRandomIntInArray);
046    }
047
048}

001import java.util.HashMap;
002import java.util.ArrayList;
003import java.util.Random;
004
005/**
006 * Thid class create a random room
007 *
008 * @author C.Diouy
009 */
010public class RoomRandomizer {
011    private HashMap<String, Room> aRooms;
012
013    /**
014     * this constructor create a room randomizer
015     * 
016     * @param pRooms Hashmap contain all the room
017     */
018    public RoomRandomizer(final HashMap<String, Room> pRooms) {
019        this.aRooms = pRooms;
020    }
021
022    /**
023     * Find a random room
024     * 
025     * @return room in array list
026     */
027    public Room findRandomRoom() {
028        Random vRandomNb = new Random();
029        ArrayList<Room> vRoomsList = new ArrayList<Room>(this.aRooms.values());
030        int vNb = vRandomNb.nextInt(vRoomsList.size());
031        return vRoomsList.get(vNb);
032    }
033}

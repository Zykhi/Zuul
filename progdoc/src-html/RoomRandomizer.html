001import java.util.HashMap;
002import java.util.Random;
003
004/**
005 * This class represents part of TransporterRoom
006 * 
007 * @author C.Diouy
008 * @version 2022.03.02
009 */
010public class RoomRandomizer {
011    private HashMap<String, Room> aAllRooms;
012    private Object[] aRoomsArrayList;
013    private boolean aAlea;
014
015    /**
016     * this constructor create a room randomizer
017     */
018    public RoomRandomizer() {
019        this.aAlea = false;
020    }
021
022    /**
023     * Find a random room
024     * 
025     * @return room in array list
026     */
027    public Room findRandomRoom() {
028        Random vRandom = new Random();
029        int vRandomIntInArray = vRandom.nextInt(this.aRoomsArrayList.length);
030        return (Room) this.aRoomsArrayList[vRandomIntInArray];
031    }
032
033    /**
034     * This method set the room for alea command
035     * 
036     * @param pRoom Room to set
037     */
038    public void setAlea(Room pRoom) {
039        this.aRoomsArrayList = new Object[] { pRoom };
040    }
041
042    /**
043     * This method set all the room random for the transporterRoom
044     * 
045     * @param pAllRooms all the room of the game
046     */
047    public void setRandom(final HashMap<String, Room> pAllRooms) {
048        this.aAllRooms = pAllRooms;
049        this.aRoomsArrayList = this.aAllRooms.values().toArray();
050    }
051
052    /**
053     * This boolean return if alea is on or off
054     * 
055     * @return true if aAlea is active
056     *         false if isnt
057     */
058    public boolean isAlea() {
059        return this.aAlea;
060    }
061
062}

001package pkg_Core;
002import pkg_UI.UserInterface;
003
004
005/**
006 * This class is the main class of the "Zuul GOTY Edition" application.
007 * * "Zuul GOTY Edition" is a very simple, text based adventure game.. Users
008 * can walk around some scenery. That's all. It should really be extended
009 * to make it more interesting!
010 * 
011 * To play this game, create an instance of this class.
012 * 
013 * This main class creates the necessary implementation objects and starts the
014 * game off.
015 * 
016 * @author Michael Kolling and David J. Barnes
017 * @version 2.0 (Jan 2003) DB edited (2019) CD edited(2022)
018 */
019
020public class Game {
021    private UserInterface aGui;
022    private GameEngine aEngine;
023
024    /**
025     * Create the game and initialise its internal map. Create the inerface and link
026     * to it.
027     */
028    public Game() {
029        this.aEngine = new GameEngine();
030        this.aGui = new UserInterface(this.aEngine);
031        this.aEngine.setGUI(this.aGui);
032    }
033}

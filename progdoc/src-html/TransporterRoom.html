001/**
002 * This class create a transporter room
003 * 
004 * @author C.Diouy
005 * @version 2022.03.02
006 */
007public class TransporterRoom extends Room {
008
009    private RoomRandomizer aRoomRandomizer;
010
011    /**
012     * Constructor of Transporter Room
013     * 
014     * @param pDescription description of the room
015     * @param pImage       image of the room
016     * @param pRoomRandomizer Room random
017     */
018    public TransporterRoom(String pDescription, String pImage, final RoomRandomizer pRoomRandomizer) {
019        super(pDescription, pImage);
020        this.aRoomRandomizer = pRoomRandomizer;
021    }
022
023    /**
024     * 
025     * @param pDirection direction of the exit
026     * @return a random room
027     */
028    @Override
029    public Room getExit(final String pDirection) {
030        return this.aRoomRandomizer.findRandomRoom();
031    }
032}

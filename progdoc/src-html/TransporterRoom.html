001import java.util.HashMap;
002
003/**
004 * This class create a transporter room
005 * 
006 * @author C.Diouy
007 * @version 2022.03.02
008 */
009public class TransporterRoom extends Room {
010
011    private RoomRandomizer aRoomRandomizer;
012
013    /**
014     * Constructor of Transporter Room
015     * 
016     * @param pDescription description of the room
017     * @param pImage       image of the room
018     */
019    public TransporterRoom(String pDescription, String pImage) {
020        super(pDescription, pImage);
021
022    }
023
024    /**
025     * This method set all the room in the room randomizer
026     * 
027     * @param pAllRooms hashmap contain all the room
028     */
029    public void setAllRooms(final HashMap<String, Room> pAllRooms) {
030        this.aRoomRandomizer = new RoomRandomizer(pAllRooms);
031    }
032
033    /**
034     * 
035     * @param pDirection direction of the exit
036     * @return a random room
037     */
038    @Override
039    public Room getExit(final String pDirection) {
040        return this.aRoomRandomizer.findRandomRoom();
041    }
042
043}

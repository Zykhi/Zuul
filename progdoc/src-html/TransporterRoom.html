001import java.util.HashMap;
002
003/**
004 * This class create a transporter room
005 * 
006 * @author C.Diouy
007 * @version 2022.03.02
008 */
009
010public class TransporterRoom extends Room {
011    private RoomRandomizer aRandomRoom;
012    private String aAlea;
013    private HashMap<String, Room> aRooms;
014
015    /**
016     * Constructor of Transporter Room
017     * 
018     * @param pDescription description of the room
019     * @param pImage       image of the room
020     * @param pRooms       hashmap contains all the rooms
021     */
022    public TransporterRoom(final String pDescription, final String pImage, final HashMap<String, Room> pRooms) {
023        super(pDescription, pImage);
024        this.aRandomRoom = new RoomRandomizer(pRooms);
025        this.aRooms = pRooms;
026        this.aAlea = null;
027    }
028
029    /**
030     * Refactior getExit method of Room class
031     * 
032     * @param pDirection direction of the exit
033     * @return a random room
034     */
035    @Override
036    public Room getExit(final String pDirection) {
037        if (this.aAlea == null) {
038            return this.aRandomRoom.findRandomRoom();
039        } else {
040            return this.aRooms.get(this.aAlea);
041        }
042    }
043
044    /**
045     * This method modify the random caracter of the random room
046     * 
047     * @param pNextRoom the room were transporter room teleport you
048     */
049    public void setNextRoom(final String pNextRoom) {
050        this.aAlea = pNextRoom;
051    }
052}

001import java.util.HashMap;
002import java.util.Set;
003
004/**
005 * Class Room - a room in an adventure game.
006 * 
007 * This class is part of the "Zuul GOTY Edition" application.
008 * "Zuul GOTY Edition" is a very simple, text based adventure game.
009 * 
010 * A "Room" represents one location in the scenery of the game. It is
011 * connected to other rooms via exits. For each existing exit, the room
012 * stores a reference to the neighboring room.
013 *
014 * @author Michael Kolling and David J. Barnes + D.Bureau + C.Diouy
015 * @version 2008.03.30 + 2019.09.25 + 2022.02.11
016 */
017public class Room {
018    private String aDescription;
019    private HashMap<String, Room> aExits;
020    private ItemList aItems;
021    private String aImageName;
022
023    /**
024     * Create a room described by "aDescription" with a given image.
025     * Initially, it has no exits. "aDescription" is something like
026     * "in a kitchen" or "in an open court yard".
027     * 
028     * @param pDescription description of the room
029     * @param pImage       image of the room
030     */
031    public Room(final String pDescription, final String pImage) {
032        this.aDescription = pDescription;
033        this.aExits = new HashMap<String, Room>();
034        this.aItems = new ItemList();
035        this.aImageName = pImage;
036    }
037
038    /**
039     * This String get a description
040     * 
041     * @return the description of the room (the one that was defined in the
042     *         constructor).
043     */
044    public String getDescription() {
045        return this.aDescription;
046    }
047
048    /**
049     * return a long description of the room
050     * 
051     * @return a long description of this room, in the form:
052     *         You are in the kitchen.
053     *         Exits: north west
054     */
055    public String getLongDescription() {
056        return "You are " + aDescription + ".\n" +
057                getExitString() + "\n" +
058                getItemString();
059    }
060
061    // EXIT
062
063    /**
064     * this Room get exit
065     * 
066     * @param pDirection name of the exit
067     * @return the room that is reached if we go from this room in direction
068     *         "direction". If there is no room in that direction, return null.
069     */
070    public Room getExit(String pDirection) {
071        return aExits.get(pDirection);
072    }
073
074    /**
075     * Define an exit from this room.
076     * 
077     * @param pDirection name of the exit
078     * @param pNeighbor  room neighbor of actual room
079     */
080    public void setExit(final String pDirection, Room pNeighbor) {
081        aExits.put(pDirection, pNeighbor);
082    }
083
084    /**
085     * This String get all the exits in the String
086     * 
087     * @return a string describing the room's exits, for example
088     *         "Exits: north west"
089     */
090    public String getExitString() {
091        String vExitString = "Exits : ";
092        Set<String> vKeys = aExits.keySet();
093        for (String vExit : vKeys) {
094            vExitString += " " + vExit;
095        }
096        return vExitString;
097    }
098
099    // ITEM
100
101    /**
102     * this Item get name
103     * 
104     * @param pName name of the item
105     * @return the name of the item
106     *         If there is no item with this name, return null.
107     */
108    public Item getItemName(String pName) {
109        return this.aItems.getItemName(pName);
110    }
111
112    /**
113     * Define an item from this room.
114     * 
115     * @param pName name of the item
116     * @param pItem variable item
117     */
118    public void addItem(final String pName, final Item pItem) {
119        this.aItems.addItem(pName, pItem);
120    }
121
122    public void removeItem(final String pName, final Item pItem) {
123        this.aItems.removeItem(pName, pItem);
124    }
125
126    /**
127     * This function get the string of the item in the room
128     * 
129     * @return the item in the room
130     */
131    public String getItemString() {
132        if (this.aItems.isEmpty()) {
133            return "No item here.";
134        } else {
135            return this.aItems.getItemString();
136        }
137    }
138
139    // IMAGE
140
141    /**
142     * This function get the image
143     * 
144     * @return a string describing the room's image name
145     */
146    public String getImageName() {
147        return this.aImageName;
148    }
149
150} // Room

150    /**
151     * Given a command, process (that is: execute) the command.
152     * 
153     * @param pCommandLine command who's enter by the player
154     */
155    public void interpretCommand(final Command pCommandLine) {
156
157        CommandWord vCommandWord = pCommandLine.getCommandWord();
158        if (aPlayer.getMovement() <= 0) {
159            this.gameOver();
160        } else {
161
162            try {
163
164                switch (vCommandWord) {
165                    case HELP:
166                        this.aPlayer.printHelp();
167                        getCurrentRoomItemsString();
168                        break;
169
170                    case GO:
171                        this.aPlayer.goRoom(pCommandLine);
172                        break;
173
174                    case QUIT:
175                        if (pCommandLine.hasSecondWord()) {
176                            this.aGui.println("Quit what?");
177                        } else {
178                            this.endGame();
179                        }
180                        break;
181
182                    case LOOK:
183                        this.aPlayer.look(pCommandLine);
184                        break;
185
186                    case EAT:
187                        this.aPlayer.eat(pCommandLine);
188                        break;
189
190                    case BACK:
191                        this.aPlayer.back(pCommandLine);
192                        break;
193
194                    case TEST:
195                        this.test(pCommandLine);
196                        break;
197
198                    case TAKE:
199                        this.aPlayer.take(pCommandLine);
200                        break;
201
202                    case DROP:
203                        this.aPlayer.drop(pCommandLine);
204                        break;
205
206                    case INVENTORY:
207                        this.aPlayer.showInventory();
208                        break;
209
210                    case CHARGE:
211                        this.aPlayer.charge();
212                        break;
213
214                    case FIRE:
215                        this.aPlayer.fire();
216                        break;
217
218                    case EXIT:
219                        this.aPlayer.exit();
220                        break;
221
222                    case SKIP:
223                        this.aGui.skipMethod();
224                        break;
225
226                    case ALEA:
227                        this.alea(pCommandLine);
228                        break;
229
230                    default:
231                        this.aGui.println("I don't know what you mean...");
232                        break;
233                }
234
235            } catch (Exception pE) {
236                // use try catch to avoid error
237            }
238        }
239    }

257    /**
258     * This function get the items in the room. This function is used for the
259     * UserInterface
260     * 
261     * @return String with all items in the room
262     */
263    protected String getCurrentRoomItemsString() {
264        String[] vCRIS = aPlayer.getCurrentRoomItemsString().split(" : ");
265        if (vCRIS.length > 1) {
266            return vCRIS[1];
267        } else {
268            return null;
269        }
270    }

280        if (vCIIS.length > 1) {

283            return null;

285    }

287    /**
288     * This method is used to test game command
289     * 
290     * @param pFile test file
291     */
292    private void test(Command pFile) {
293        if (pFile.hasSecondWord()) {
294            try {
295                String vFile = pFile.getSecondWord();
296                Scanner vScanner = new Scanner(new File(vFile + ".txt"));
297                String vCommandString = vScanner.nextLine();
298                this.aTest = true;
299                while (vScanner.hasNextLine()) {
300                    Command vCommand = aParser.getCommand(vCommandString);
301                    interpretCommand(vCommand);
302                    vCommandString = vScanner.nextLine();
303                }
304                this.aTest = false;
305            } catch (final FileNotFoundException pE) {
306                this.aGui.println(pE.getMessage());
307            }
308        } else {
309            this.aGui.println("This command need a second word");
310        }
311    }

313    /**
314     * this method is used to set a room for the random room
315     * 
316     * @param pRoom Room to set for the exit of the random room
317     */
318    private void alea(final Command pRoom) {
319        if (!aTest) {
320            this.aGui.println("You need to be in test mode");
321        }
322        if (pRoom.hasSecondWord() && !aRandomRoom.isAlea()) {
323            Room vRoom = aRooms.get(pRoom.getSecondWord());
324            if (vRoom == null) {
325                System.out.println("Room dont found");
326            }
327            aRandomRoom.setAlea(vRoom);
328        } else {
329            aRandomRoom.setRandom(aRooms);
330        }
331    }

006/**
007 * This class is part of the "Zuul GOTY Edition" application.
008 * "Zuul GOTY Edition" is a very simple, text based adventure game.

126    /**

128     */

136    /**
137     * Given a command, process (that is: execute) the command.
138     * 
139     * @param pCommandLine command who's enter by the player
140     */
141    public void interpretCommand(final Command pCommandLine) {
142
143        this.aGui.println("> " + aGui.getEntryField());
144
145        CommandWord vCommandWord = pCommandLine.getCommandWord();
146
147        try {
148
149            switch (vCommandWord) {
150                case HELP:
151                    this.aPlayer.printHelp();
152                    break;
153
154                case GO:
155                    this.aPlayer.goRoom(pCommandLine);
156                    break;
157
158                case QUIT:
159                    if (pCommandLine.hasSecondWord()) {
160                        this.aGui.println("Quit what?");
161                    } else {
162                        this.endGame();
163                    }
164                    break;
165
166                case LOOK:
167                    this.aPlayer.look(pCommandLine);
168                    break;
169
170                case EAT:
171                    this.aPlayer.eat(pCommandLine);
172                    break;
173
174                case BACK:
175                    this.aPlayer.back(pCommandLine);
176                    break;
177
178                case TEST:
179                    this.test(pCommandLine);
180                    break;
181
182                case TAKE:
183                    this.aPlayer.take(pCommandLine);
184                    break;
185
186                case DROP:
187                    this.aPlayer.drop(pCommandLine);
188                    break;
189
190                case INVENTORY:
191                    this.aPlayer.showInventory();
192                    break;
193
194                default:
195                    this.aGui.println("I don't know what you mean...");
196                    break;
197            }
198
199        } catch (Exception pE) {
200            // use try catch to avoid error
201            // System.out.println(pE.getMessage());
202        }
203    }
204
205    /**
206     * End the game
207     */
208    private void endGame() {
209        this.aGui.println("Thank you for playing.  Good bye.");
210        this.aGui.enable(false);
211    }

213    /**
214     * This method is used to test game command
215     * 
216     * @param pFile test file
217     */
218    private void test(Command pFile) {
219        if (pFile.hasSecondWord()) {
220            try {
221                String vFile = pFile.getSecondWord();
222                Scanner vScanner = new Scanner(new File(vFile + ".txt"));
223                String vCommandString = vScanner.nextLine();
224                while (vScanner.hasNextLine()) {
225                    Command vCommand = aParser.getCommand(vCommandString);
226                    interpretCommand(vCommand);
227                    vCommandString = vScanner.nextLine();
228                }
229            } catch (final FileNotFoundException pE) {
230                this.aGui.println(pE.getMessage());
231            }
232        } else {
233            this.aGui.println("This command need a second word");
234        }
235    }

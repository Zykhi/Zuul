151    /**
152     * Given a command, process (that is: execute) the command.
153     * 
154     * @param pCommandLine command who's enter by the player
155     */
156    public void interpretCommand(final Command pCommandLine) {
157
158        CommandWord vCommandWord = pCommandLine.getCommandWord();
159        if (aPlayer.getMovement() <= 0) {
160            this.gameOver();
161        }
162        if (this.aGui.isTimerEnd()) {
163            this.gameOver();
164            this.aGui.stopTimer();
165        } else {
166
167            try {
168
169                switch (vCommandWord) {
170                    case HELP:
171                        this.aPlayer.printHelp();
172                        getCurrentRoomItemsString();
173                        break;
174
175                    case GO:
176                        this.aPlayer.goRoom(pCommandLine);
177                        break;
178
179                    case QUIT:
180                        if (pCommandLine.hasSecondWord()) {
181                            this.aGui.println("Quit what?");
182                        } else {
183                            this.endGame();
184                        }
185                        break;
186
187                    case LOOK:
188                        this.aPlayer.look(pCommandLine);
189                        break;
190
191                    case EAT:
192                        this.aPlayer.eat(pCommandLine);
193                        break;
194
195                    case BACK:
196                        this.aPlayer.back(pCommandLine);
197                        break;
198
199                    case TEST:
200                        this.test(pCommandLine);
201                        break;
202
203                    case TAKE:
204                        this.aPlayer.take(pCommandLine);
205                        break;
206
207                    case DROP:
208                        this.aPlayer.drop(pCommandLine);
209                        break;
210
211                    case INVENTORY:
212                        this.aPlayer.showInventory();
213                        break;
214
215                    case CHARGE:
216                        this.aPlayer.charge();
217                        break;
218
219                    case FIRE:
220                        this.aPlayer.fire();
221                        break;
222
223                    case EXIT:
224                        this.aPlayer.exit();
225                        break;
226
227                    case SKIP:
228                        this.aGui.skipMethod();
229                        break;
230
231                    case ALEA:
232                        this.alea(pCommandLine);
233                        break;
234
235                    default:
236                        this.aGui.println("I don't know what you mean...");
237                        break;
238                }
239
240            } catch (Exception pE) {
241                // use try catch to avoid error
242            }
243        }
244    }

262    /**
263     * This function get the items in the room. This function is used for the
264     * UserInterface
265     * 
266     * @return String with all items in the room
267     */
268    protected String getCurrentRoomItemsString() {
269        String[] vCRIS = aPlayer.getCurrentRoomItemsString().split(" : ");
270        if (vCRIS.length > 1) {
271            return vCRIS[1];
272        } else {
273            return null;
274        }
275    }

285        if (vCIIS.length > 1) {

292    /**
293     * This method is used to test game command
294     * 
295     * @param pFile test file
296     */
297    private void test(Command pFile) {
298        if (pFile.hasSecondWord()) {
299            try {
300                String vFile = pFile.getSecondWord();
301                Scanner vScanner = new Scanner(new File("testFiles/" + vFile + ".txt"));
302                String vCommandString = vScanner.nextLine();
303                this.aTest = true;
304                while (vScanner.hasNextLine()) {
305                    Command vCommand = aParser.getCommand(vCommandString);
306                    interpretCommand(vCommand);
307                    vCommandString = vScanner.nextLine();
308                }
309                this.aTest = false;
310            } catch (final FileNotFoundException pE) {
311                this.aGui.println(pE.getMessage());
312            }
313        } else {
314            this.aGui.println("This command need a second word");
315        }
316    }

318    /**
319     * this method is used to set a room for the random room
320     * 
321     * @param pRoom Room to set for the exit of the random room
322     */
323    private void alea(final Command pRoom) {
324        if (!aTest) {
325            this.aGui.println("You need to be in test mode");
326        }
327        if (pRoom.hasSecondWord() && aTest) {
328            String vRoomName = pRoom.getSecondWord();
329            if (vRoomName == null) {
330                System.out.println("Room dont found");
331            }
332            for (TransporterRoom vTransporterRoom : this.aTransporterRoom) {
333                vTransporterRoom.setNextRoom(vRoomName);
334            }
335        }
336    }

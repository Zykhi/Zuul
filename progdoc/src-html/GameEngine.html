153    /**
154     * Given a command, process (that is: execute) the command.
155     * 
156     * @param pCommandLine command who's enter by the player
157     */
158    public void interpretCommand(final Command pCommandLine) {
159
160        CommandWord vCommandWord = pCommandLine.getCommandWord();
161        if (aPlayer.getMovement() <= 0) {
162            this.gameOver();
163        } else if (this.aGui.isTimerEnd()) {
164            this.gameOver();
165            this.aGui.stopTimer();
166        } else {
167
168            try {
169
170                switch (vCommandWord) {
171                    case HELP:
172                        this.aPlayer.printHelp();
173                        getCurrentRoomItemsString();
174                        break;
175
176                    case GO:
177                        this.aPlayer.goRoom(pCommandLine);
178                        break;
179
180                    case QUIT:
181                        if (pCommandLine.hasSecondWord()) {
182                            this.aGui.println("Quit what?");
183                        } else {
184                            this.endGame();
185                        }
186                        break;
187
188                    case LOOK:
189                        this.aPlayer.look(pCommandLine);
190                        break;
191
192                    case EAT:
193                        this.aPlayer.eat(pCommandLine);
194                        break;
195
196                    case BACK:
197                        this.aPlayer.back(pCommandLine);
198                        break;
199
200                    case TEST:
201                        this.test(pCommandLine);
202                        break;
203
204                    case TAKE:
205                        this.aPlayer.take(pCommandLine);
206                        break;
207
208                    case DROP:
209                        this.aPlayer.drop(pCommandLine);
210                        break;
211
212                    case INVENTORY:
213                        this.aPlayer.showInventory();
214                        break;
215
216                    case CHARGE:
217                        this.aPlayer.charge();
218                        break;
219
220                    case FIRE:
221                        this.aPlayer.fire();
222                        break;
223
224                    case EXIT:
225                        this.aPlayer.exit();
226                        break;
227
228                    case SKIP:
229                        this.aGui.skipMethod();
230                        break;
231
232                    case ALEA:
233                        this.alea(pCommandLine);
234                        break;
235
236                    default:
237                        this.aGui.println("I don't know what you mean...");
238                        break;
239                }
240
241            } catch (Exception pE) {
242                // use try catch to avoid error
243            }
244        }
245    }

263    /**
264     * This function get the items in the room. This function is used for the
265     * UserInterface
266     * 
267     * @return String with all items in the room
268     */
269    protected String getCurrentRoomItemsString() {
270        String[] vCRIS = aPlayer.getCurrentRoomItemsString().split(" : ");
271        if (vCRIS.length > 1) {
272            return vCRIS[1];
273        } else {
274            return null;
275        }
276    }

293    /**
294     * This function get the key of the room with the value of the HashMap
295     * 
296     * @param pHashMap the HashMap
297     * @param pRoom    the room in value in the HashMap
298     * @return String of the key of the room
299     * @author https://www.techiedelight.com/get-map-key-from-value-java/
300     */
301    private String getKey(HashMap<String, Room> pHashMap, Room pRoom) {
302        for (String vKey : pHashMap.keySet()) {
303            if (pRoom.equals(pHashMap.get(vKey))) {
304                return vKey;
305            }
306        }
307        return null;
308    }

310    /**
311     * This method play the sound of the current room
312     */
313    public void playRoomSound() {
314        HashMap<String, Room> vRooms = aRooms;
315        Room vRoom = aPlayer.getCurrentRoom();
316        String vCurrentRoomString = getKey(vRooms, vRoom);
317        this.aGui.stopSound();
318        this.aGui.playSound(vCurrentRoomString, -1);
319    }

321    /**
322     * This method is used to test game command
323     * 
324     * @param pFile test file
325     */
326    private void test(Command pFile) {
327        if (pFile.hasSecondWord()) {
328            try {
329                String vFile = pFile.getSecondWord();
330                Scanner vScanner = new Scanner(new File("testFiles/" + vFile + ".txt"));
331                String vCommandString = vScanner.nextLine();
332                this.aTest = true;
333                while (vScanner.hasNextLine()) {
334                    Command vCommand = aParser.getCommand(vCommandString);
335                    interpretCommand(vCommand);
336                    vCommandString = vScanner.nextLine();
337                }
338                this.aTest = false;
339            } catch (final FileNotFoundException pE) {
340                this.aGui.println(pE.getMessage());
341            }
342        } else {
343            this.aGui.println("This command need a second word");
344        }
345    }

347    /**
348     * this method is used to set a room for the random room
349     * 
350     * @param pRoom Room to set for the exit of the random room
351     */
352    private void alea(final Command pRoom) {
353        if (!aTest) {
354            this.aGui.println("You need to be in test mode");
355        }
356        if (pRoom.hasSecondWord() && aTest) {
357            String vRoomName = pRoom.getSecondWord();
358            if (vRoomName == null) {
359                System.out.println("Room dont found");
360            }
361            for (TransporterRoom vTransporterRoom : this.aTransporterRoom) {
362                vTransporterRoom.setNextRoom(vRoomName);
363            }
364        }

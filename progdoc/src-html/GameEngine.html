007/**

158    /**
159     * Given a command, process (that is: execute) the command.
160     * 
161     * @param pCommandLine command who's enter by the player
162     */
163    public void interpretCommand(final Command pCommandLine) {
164
165        this.aGui.println("> " + aGui.getEntryField());
166
167        CommandWord vCommandWord = pCommandLine.getCommandWord();
168        if (aPlayer.getMovement() <= 0) {
169            this.gameOver();
170        } else {
171
172            try {
173
174                switch (vCommandWord) {
175                    case HELP:
176                        this.aPlayer.printHelp();
177                        break;
178
179                    case GO:
180                        this.aPlayer.goRoom(pCommandLine);
181                        break;
182
183                    case QUIT:
184                        if (pCommandLine.hasSecondWord()) {
185                            this.aGui.println("Quit what?");
186                        } else {
187                            this.endGame();
188                        }
189                        break;
190
191                    case LOOK:
192                        this.aPlayer.look(pCommandLine);
193                        break;
194
195                    case EAT:
196                        this.aPlayer.eat(pCommandLine);
197                        break;
198
199                    case BACK:
200                        this.aPlayer.back(pCommandLine);
201                        break;
202
203                    case TEST:
204                        this.test(pCommandLine);
205                        break;
206
207                    case TAKE:
208                        this.aPlayer.take(pCommandLine);
209                        break;
210
211                    case DROP:
212                        this.aPlayer.drop(pCommandLine);
213                        break;
214
215                    case INVENTORY:
216                        this.aPlayer.showInventory();
217                        break;
218
219                    case CHARGE:
220                        this.aPlayer.charge();
221                        break;
222
223                    case FIRE:
224                        this.aPlayer.fire();
225                        break;
226
227                    default:
228                        this.aGui.println("I don't know what you mean...");
229                        break;
230                }
231
232            } catch (Exception pE) {
233                // use try catch to avoid error
234                // System.out.println(pE.getMessage());
235            }
236        }
237    }

253    }
254
255    /**
256     * This method is used to test game command
257     * 
258     * @param pFile test file
259     */
260    private void test(Command pFile) {
261        if (pFile.hasSecondWord()) {
262            try {
263                String vFile = pFile.getSecondWord();
264                Scanner vScanner = new Scanner(new File(vFile + ".txt"));
265                String vCommandString = vScanner.nextLine();
266                while (vScanner.hasNextLine()) {
267                    Command vCommand = aParser.getCommand(vCommandString);
268                    interpretCommand(vCommand);
269                    vCommandString = vScanner.nextLine();
270                }
271            } catch (final FileNotFoundException pE) {
272                this.aGui.println(pE.getMessage());
273            }
274        } else {
275            this.aGui.println("This command need a second word");
276        }
277    }

001package pkg_UI;
002
003import java.awt.Color;
004import java.awt.Component;
005import java.awt.Graphics;
006import java.awt.Graphics2D;
007import java.awt.Insets;
008import java.awt.RenderingHints;
009import javax.swing.border.Border;
010
011/**
012 * @author Twinuts
013 *         https://codes-sources.commentcamarche.net/forum/affich-857479-bordure-arrondie-dans-un-jpanel
014 */
015public class HealthPanelBorder implements Border {
016
017    /**
018     * 
019     */
020    public Insets getBorderInsets(Component c) {
021        return new Insets(0, 0, 0, 0);// ou autre chose cela dépend de si tu veux rendre parametrable
022    }
023
024    /**
025     *
026     */
027    public boolean isBorderOpaque() {
028        return false;// ou autre chose cela dépend de si tu veux rendre parametrable
029    }
030
031    /**
032     *
033     */
034    public void paintBorder(Component c, Graphics g, int x, int y, int width, int height) {
035        g.setColor(Color.darkGray);// ou une autre couleur que tu peux rendre paramétrable
036        int arc = 25;// tu peux aussi le rendre configurable
037        int adjustXY = 1;// pour ajuster le dessin en x et y
038        int adjustWH = 2;// idem pour width et height
039        // pour eviter les escalier sur l'arrondi
040        Graphics2D g2 = (Graphics2D) g;
041        g2.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
042        g2.fillRoundRect(x + adjustXY, y + adjustXY, width - adjustWH, height - adjustWH, arc, arc);
043    }
044}

189    /**
190     * This method prints slowly the text in param
191     * 
192     * This code was found on internet and update for my use
193     * 
194     * @author https://github.com/Vourem/SlowPrint-Method/blob/master/SlowPrint
195     * @param pText Text you want to write slowly
196     */
197    private void slowPrint(final String pText) {
198
199        aTextTimer = new Timer(aTime, new ActionListener() {
200            @Override
201            public void actionPerformed(ActionEvent pEvent) {
202                aLog.setText(pText.substring(0, aIndex));
203                aIndex++;
204                if (aIndex > pText.length()) {
205                    ((Timer) pEvent.getSource()).stop();
206                }
207            }
208
209        });
210
211        aTextTimer.start();
212        aIndex = 0;
213    }

609    /**
610     * Enable or disable input in the input field.
611     * 
612     * @param pOnOff true when enable
613     *               false when disable
614     */
615    public void enable(final boolean pOnOff) {
616        this.aEntryField.setEditable(pOnOff); // enable/disable
617        if (!pOnOff) { // disable
618            this.aEntryField.getCaret().setBlinkRate(0); // cursor won't blink
619            this.aEntryField.removeActionListener(this); // won't react to entry
620
621            JButton[] vButtons = { aBackButton, aHelpButton, aQuitButton, aDropButton, aTakeButton, aGiveButton,
622                    aChargeButton, aInventoryButton, aNorthButton, aSouthButton, aEastButton, aWestButton, aUpButton,
623                    aDownButton, aBattleButton };
624
625            for (JButton currentButton : vButtons) {
626                for (ActionListener al : currentButton.getActionListeners()) {
627                    currentButton.removeActionListener(al);
628                }
629            }
630        } else {
631            this.aEntryField.addActionListener(this);
632            this.aQuitButton.addActionListener(this);
633            this.aNorthButton.addActionListener(this);
634            this.aSouthButton.addActionListener(this);
635            this.aEastButton.addActionListener(this);
636            this.aWestButton.addActionListener(this);
637            this.aUpButton.addActionListener(this);
638            this.aDownButton.addActionListener(this);
639            this.aBackButton.addActionListener(this);
640            this.aHelpButton.addActionListener(this);
641            this.aDropButton.addActionListener(e -> dropButtonMethod());
642            this.aTakeButton.addActionListener(e -> takeButtonMethod());
643            this.aGiveButton.addActionListener(e -> giveButtonMethod());
644            this.aChargeButton.addActionListener(this);
645            this.aInventoryButton.addActionListener(this);
646            this.aBattleButton.addActionListener(this);
647        }
648    } // enable(.)

659            aFont = Font.createFont(Font.TRUETYPE_FONT, new File("font/8bit.ttf"));

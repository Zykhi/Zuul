163    /**
164     * This method prints slowly the text in param
165     * 
166     * This code was found on internet and update for my use
167     * 
168     * @author https://github.com/Vourem/SlowPrint-Method/blob/master/SlowPrint
169     * @param pText Text you want to write slowly
170     */
171    private void slowPrint(final String pText) {
172
173        aTextTimer = new Timer(aTime, new ActionListener() {
174            @Override
175            public void actionPerformed(ActionEvent pEvent) {
176                aLog.setText(pText.substring(0, aIndex));
177                aIndex++;
178                if (aIndex > pText.length()) {
179                    ((Timer) pEvent.getSource()).stop();
180                }
181            }
182
183        });
184
185        aTextTimer.start();
186        aIndex = 0;
187    }

278     * This method play sound.
279     * 
280     * This code was found on internet and update for my use
281     * 
282     * @author https://stackoverflow.com/questions/6045384/playing-mp3-and-wav-in-java
283     * @param pFile File you want to play
284     * @param pLoop Number of repetitions of the sound (-1 is infinite)
285     */
286    public void playSound(final String pFile, final int pLoop) {
287        try {
288            AudioInputStream vAudioInputStream = AudioSystem
289                    .getAudioInputStream(new File("gameSounds/" + pFile + ".wav").getAbsoluteFile());
290            aClip = AudioSystem.getClip();
291            aClip.open(vAudioInputStream);
292            aClip.start();
293            aClip.loop(pLoop);
294        } catch (Exception ex) {
295            System.out.println("Error with playing sound.");
296            ex.printStackTrace();
297        }
298    }

536    /**
537     * Enable or disable input in the input field.
538     * 
539     * @param pOnOff true when enable
540     *               false when disable
541     */
542    public void enable(final boolean pOnOff) {
543        this.aEntryField.setEditable(pOnOff); // enable/disable
544        if (!pOnOff) { // disable
545            this.aEntryField.getCaret().setBlinkRate(0); // cursor won't blink
546            this.aEntryField.removeActionListener(this); // won't react to entry
547
548            JButton[] vButtons = { aBackButton, aHelpButton, aQuitButton, aDropButton, aTakeButton, aFireButton,
549                    aChargeButton, aInventoryButton, aNorthButton, aSouthButton, aEastButton, aWestButton, aUpButton,
550                    aDownButton, aBattleButton };
551
552            for (JButton currentButton : vButtons) {
553                for (ActionListener al : currentButton.getActionListeners()) {
554                    currentButton.removeActionListener(al);
555                }
556            }
557        } else {
558            this.aEntryField.addActionListener(this);
559            this.aQuitButton.addActionListener(this);
560            this.aNorthButton.addActionListener(this);
561            this.aSouthButton.addActionListener(this);
562            this.aEastButton.addActionListener(this);
563            this.aWestButton.addActionListener(this);
564            this.aUpButton.addActionListener(this);
565            this.aDownButton.addActionListener(this);
566            this.aBackButton.addActionListener(this);
567            this.aHelpButton.addActionListener(this);
568            this.aDropButton.addActionListener(e -> dropButtonMethod());
569            this.aTakeButton.addActionListener(e -> takeButtonMethod());
570            this.aFireButton.addActionListener(this);
571            this.aChargeButton.addActionListener(this);
572            this.aInventoryButton.addActionListener(this);
573            this.aBattleButton.addActionListener(this);
574        }
575    } // enable(.)

579    /**
580     * This method create custom font
581     */
582    private void createFont() {
583        // to import custom font
584        // https://www.ryisnow.online/2021/04/java-for-beginner-how-to-use-custom-font.html
585        try {
586            aFont = Font.createFont(Font.TRUETYPE_FONT, new File("font/8bit.ttf"));
587            GraphicsEnvironment vGraphicsEnvironment = GraphicsEnvironment.getLocalGraphicsEnvironment();
588            vGraphicsEnvironment.registerFont(aFont);
589
590            aTextFont = aFont.deriveFont(18f);
591            aButtonsFont = aFont.deriveFont(14f);
592            aBattleFont = aFont.deriveFont(24f);
593            aMenuFont = aFont.deriveFont(25f);
594        } catch (IOException | FontFormatException e) {
595            e.printStackTrace();
596        }
597    }

1346     * @param pClip   the sound clip
1347     */
1348    // https://stackoverflow.com/questions/40514910/set-volume-of-java-clip
1349    private void setVolume(float pVolume, Clip pClip) {
1350        if (pVolume < 0f || pVolume > 1f)
1351            throw new IllegalArgumentException("Volume not valid: " + pVolume);
1352        FloatControl gainControl = (FloatControl) pClip.getControl(FloatControl.Type.MASTER_GAIN);
1353        gainControl.setValue(20f * (float) Math.log10(pVolume));
1354    }

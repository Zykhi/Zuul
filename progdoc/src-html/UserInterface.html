001import java.net.URL;
002
003import java.awt.event.ActionListener;
004import java.awt.event.ActionEvent;
005import java.awt.event.WindowAdapter;
006import java.awt.event.WindowEvent;
007
008import java.awt.BorderLayout;
009import java.awt.Dimension;
010
011import javax.swing.ImageIcon;
012import javax.swing.JFrame;
013import javax.swing.JLabel;
014import javax.swing.JPanel;
015import javax.swing.JScrollPane;
016import javax.swing.JTextArea;
017import javax.swing.JTextField;
018import javax.swing.JButton;
019
020/**
021 * This class implements a simple graphical user interface with a text entry
022 * area, a text output area and an optional image.
023 * 
024 * @author Michael Kolling
025 * @version 1.0 (Jan 2003) DB edited (2019) CD edited (2022)
026 */
027public class UserInterface implements ActionListener {
028    private GameEngine aEngine;
029    private JFrame aMyFrame;
030    private JTextField aEntryField;
031    private JTextArea aLog;
032    private JLabel aImage;
033    private JButton aButton;
034    private Parser aParser;
035
036    /**
037     * Construct a UserInterface. As a parameter, a Game Engine
038     * (an object processing and executing the game commands) is
039     * needed.
040     * 
041     * @param pGameEngine The GameEngine object implementing the game logic.
042     */
043    public UserInterface(final GameEngine pGameEngine) {
044        this.aEngine = pGameEngine;
045        this.createGUI();
046        this.aParser = new Parser();
047    } // UserInterface(.)
048
049    /**
050     * Print out some text into the text area.
051     * 
052     * @param pText like sysout but for gui, is the text in " "
053     */
054    public void print(final String pText) {
055        this.aLog.append(pText);
056        this.aLog.setCaretPosition(this.aLog.getDocument().getLength());
057    } // print(.)
058
059    /**
060     * Print out some text into the text area, followed by a line break.
061     * 
062     * @param pText like sysout but for gui, is the text in " "
063     */
064    public void println(final String pText) {
065        this.print(pText + "\n");
066    } // println(.)
067
068    /**
069     * Show an image file in the interface.
070     * 
071     * @param pImageName name of the image for the room
072     */
073    public void showImage(final String pImageName) {
074        String vImagePath = "" + pImageName; // to change the directory
075        URL vImageURL = this.getClass().getClassLoader().getResource(vImagePath);
076        if (vImageURL == null)
077            System.out.println("Image not found : " + vImagePath);
078        else {
079            ImageIcon vIcon = new ImageIcon(vImageURL);
080            this.aImage.setIcon(vIcon);
081            this.aMyFrame.pack();
082        }
083    } // showImage(.)
084
085    /**
086     * Enable or disable input in the input field.
087     * 
088     * @param pOnOff true when enable
089     *               false when disable
090     */
091    public void enable(final boolean pOnOff) {
092        this.aEntryField.setEditable(pOnOff); // enable/disable
093        if (!pOnOff) { // disable
094            this.aEntryField.getCaret().setBlinkRate(0); // cursor won't blink
095            this.aEntryField.removeActionListener(this); // won't react to entry
096        }
097    } // enable(.)
098
099    /**
100     * Set up graphical user interface.
101     */
102    private void createGUI() {
103        this.aMyFrame = new JFrame("Zuul GOTY Edition"); // change the title
104        this.aEntryField = new JTextField(34);
105        this.aButton = new JButton("quit");
106
107        this.aLog = new JTextArea();
108        this.aLog.setEditable(false);
109        JScrollPane vListScroller = new JScrollPane(this.aLog);
110        vListScroller.setPreferredSize(new Dimension(200, 200));
111        vListScroller.setMinimumSize(new Dimension(100, 100));
112
113        JPanel vPanel = new JPanel();
114        this.aImage = new JLabel();
115
116        vPanel.setLayout(new BorderLayout()); // ==> only five places
117        vPanel.add(this.aImage, BorderLayout.NORTH);
118        vPanel.add(vListScroller, BorderLayout.CENTER);
119        vPanel.add(this.aEntryField, BorderLayout.SOUTH);
120        vPanel.add(this.aButton, BorderLayout.WEST);
121
122        this.aMyFrame.getContentPane().add(vPanel, BorderLayout.CENTER);
123
124        // add some event listeners to some components
125        this.aEntryField.addActionListener(this);
126        this.aButton.addActionListener(this);
127
128        // to end program when window is closed
129        this.aMyFrame.addWindowListener(new WindowAdapter() {
130            public void windowClosing(WindowEvent e) {
131                System.exit(0);
132            }
133        });
134
135        this.aMyFrame.pack();
136        this.aMyFrame.setVisible(true);
137        this.aEntryField.requestFocus();
138    } // createGUI()
139
140    /**
141     * Actionlistener interface for entry textfield.
142     * 
143     * @param pE event of actionListener
144     */
145    public void actionPerformed(final ActionEvent pE) {
146        // check the type of action
147        if (pE.getActionCommand() != null) {
148            this.aEngine.interpretCommand(aParser.getCommand(pE.getActionCommand()));
149            this.aEntryField.setText(""); // to reset entry field
150        } else {
151            this.processCommand(); // never suppress this line
152        }
153    } // actionPerformed(.)
154
155    /**
156     * A command has been entered. Read the command and do whatever is
157     * necessary to process it.
158     */
159    private void processCommand() {
160        String vInput = this.aEntryField.getText();
161        this.aEntryField.setText("");
162
163        this.aEngine.interpretCommand(aParser.getCommand(vInput));
164    } // processCommand()
165
166    /**
167     * get the entry field
168     * @return the text in the entry field
169     */
170    public String getEntryField() {
171        return this.aEntryField.getText();
172    }
173} // UserInterface

059    /**
060     * Print out some text into the text area.
061     * 
062     * @param pText like sysout but for gui, is the text in " "
063     */
064    public void print(final String pText) {
065        this.aLog.append(pText);
066        this.aLog.setCaretPosition(this.aLog.getDocument().getLength());
067    } // print(.)

069    /**
070     * Print out some text into the text area, followed by a line break.
071     * 
072     * @param pText like sysout but for gui, is the text in " "
073     */
074    public void println(final String pText) {
075        this.print(pText + "\n");
076    } // println(.)

078    /**
079     * This method prints slowly the text in param
080     * 
081     * This code was found on internet and update for my use
082     * 
083     * @author https://github.com/Vourem/SlowPrint-Method/blob/master/SlowPrint
084     * @param pText Text you want to write slowly
085     */
086    public void slowPrint(final String pText) {
087
088        for (char c : pText.toCharArray()) {
089            String vMessage = Character.toString(c);
090
091            this.aLog.append(vMessage);
092            this.aLog.setCaretPosition(this.aLog.getDocument().getLength());
093
094            try {
095                Thread.sleep(aTimer);
096            } catch (InterruptedException e) {
097                e.printStackTrace();
098            }
099        }
100    }
101
102    /**
103     * This method is like SlowPrint but for println
104     * 
105     * @param pText Text you want to write slowly
106     */
107    public void slowPrintln(final String pText) {
108        this.slowPrint(pText + "\n");
109    }

111    /**
112     * This method play sound.
113     * 
114     * This code was found on internet and update for my use
115     * 
116     * @author https://stackoverflow.com/questions/6045384/playing-mp3-and-wav-in-java
117     * @param pFile File you want to play
118     */
119    public void playSound(final String pFile) {
120        try {
121            AudioInputStream vAudioInputStream = AudioSystem
122                    .getAudioInputStream(new File("gameSounds/" + pFile + ".wav").getAbsoluteFile());
123            aClip = AudioSystem.getClip();
124            aClip.open(vAudioInputStream);
125            aClip.start();
126        } catch (Exception ex) {
127            System.out.println("Error with playing sound.");
128            ex.printStackTrace();
129        }
130    }

133     * This method stop sound.
134     */

137    }

338    /**
339     * Actionlistener interface for entry textfield.
340     * 
341     * @param pE event of actionListener
342     */
343    public void actionPerformed(final ActionEvent pE) {
344        // check the type of action
345        if (pE.getActionCommand() != null) {
346            this.aEngine.interpretCommand(aParser.getCommand(pE.getActionCommand()));
347            this.aEntryField.setText(""); // to reset entry field
348        } else {
349            this.processCommand(); // never suppress this line
350        }
351    } // actionPerformed(.)

353    /**
354     * This method is called when you click on the drop button
355     * It's to change UI to show items in inventory
356     */
357    public void dropButtonMethod() {
358        JButton[] vButtons = { aBackButton, aHelpButton, aQuitButton, aDropButton, aTakeButton, aFireButton,
359                aChargeButton, aInventoryButton };
360
361        if (aEngine.getCurrentInventoryItemsString() != null) {
362            String[] vOutput = aEngine.getCurrentInventoryItemsString().split(" ");
363            for (int i = 0; i < vOutput.length; i++) {
364                vButtons[i].setText(vOutput[i]);
365                vButtons[i].setActionCommand("drop " + vOutput[i]);
366            }
367            for (int i = vOutput.length; i < 8; i++) {
368                vButtons[i].setText("");
369                vButtons[i].setActionCommand("");
370                vButtons[i].removeActionListener(this);
371            }
372            aSkipButton.setText("exit");
373            aSkipButton.setActionCommand("exit");
374            aSkipButton.addActionListener(e -> exitButtonMethod());
375            for (int i = vOutput.length; i < 8; i++) {
376                vButtons[i].addActionListener(this);
377            }
378        }
379    }

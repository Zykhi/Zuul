088    /**
089     * This method prints slowly the text in param
090     * 
091     * This code was found on internet and update for my use
092     * 
093     * @author https://github.com/Vourem/SlowPrint-Method/blob/master/SlowPrint
094     * @param pText Text you want to write slowly
095     */
096    public void slowPrint(final String pText) {
097
098        for (char c : pText.toCharArray()) {
099            String vMessage = Character.toString(c);
100
101            this.aLog.append(vMessage);
102            this.aLog.setCaretPosition(this.aLog.getDocument().getLength());
103
104            try {
105                Thread.sleep(aTime);
106            } catch (InterruptedException e) {
107                e.printStackTrace();
108            }
109        }
110    }

117    public void slowPrintln(final String pText) {

121    /**
122     * This method play sound.
123     * 
124     * This code was found on internet and update for my use
125     * 
126     * @author https://stackoverflow.com/questions/6045384/playing-mp3-and-wav-in-java
127     * @param pFile File you want to play
128     * @param pLoop Number of repetitions of the sound (-1 is infinite)
129     */
130    public void playSound(final String pFile, final int pLoop) {
131        try {
132            AudioInputStream vAudioInputStream = AudioSystem
133                    .getAudioInputStream(new File("gameSounds/" + pFile + ".wav").getAbsoluteFile());
134            aClip = AudioSystem.getClip();
135            aClip.open(vAudioInputStream);
136            aClip.start();
137            aClip.loop(pLoop);
138        } catch (Exception ex) {
139            System.out.println("Error with playing sound.");
140            ex.printStackTrace();
141        }
142    }

261        // check the type of action
262        if (pE.getActionCommand() != null) {
263            this.aEngine.interpretCommand(aParser.getCommand(pE.getActionCommand()));
264            this.aEntryField.setText(""); // to reset entry field
265        } else {

407    /**
408     * This method is called when you click on the drop button
409     * It's to change UI to show items in inventory
410     */
411    public void dropButtonMethod() {
412        JButton[] vButtons = { aBackButton, aHelpButton, aQuitButton, aDropButton, aTakeButton, aFireButton,
413                aChargeButton, aInventoryButton };
414        if (aEngine.getCurrentInventoryItemsString() != null) {
415            String[] vOutput = aEngine.getCurrentInventoryItemsString().split(" ");
416            for (int i = 0; i < vOutput.length; i++) {
417                vButtons[i].setText(vOutput[i]);
418                vButtons[i].setActionCommand("drop " + vOutput[i]);
419            }
420            for (int i = vOutput.length; i < 8; i++) {
421                vButtons[i].setText("");
422                vButtons[i].setActionCommand("");
423                vButtons[i].removeActionListener(this);
424            }
425            aSkipButton.setText("exit");
426            aSkipButton.setActionCommand("exit");
427            aSkipButton.addActionListener(e -> exitButtonMethod());
428            for (int i = vOutput.length; i < 8; i++) {
429                vButtons[i].addActionListener(this);
430            }
431            this.println("What do you want to drop ?");
432        } else {
433            this.println("Your inventory is empty");
434        }
435    }

523    /**
524     * This method start the timer
525     * 
526     * @author https://www.developpez.net/forums/anocode.php?id=f5b514a1cac1a7a4fd52f16767094ab1
527     */
528    public void startTimer() {
529        aSecond = 0;
530        aMinute = 0;
531        aTaskTimer = new ActionListener() {
532            public void actionPerformed(ActionEvent pE) {
533
534                aSecond++;
535                if (aSecond == 60) {
536                    aSecond = 0;
537                    aMinute++;
538                }
539                aGameTimer.setText("elapsed time : " + aMinute + " : " + aSecond);
540            }
541        };
542
543        aTimer = new Timer(aDelay, aTaskTimer);
544        aTimer.start();
545    }

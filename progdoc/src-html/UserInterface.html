001import java.net.URL;
002
003import java.awt.event.ActionListener;
004import java.awt.event.ActionEvent;
005import java.awt.event.WindowAdapter;
006import java.awt.event.WindowEvent;
007
008import java.awt.BorderLayout;
009import java.awt.Dimension;
010
011import javax.swing.ImageIcon;
012import javax.swing.JFrame;
013import javax.swing.JLabel;
014import javax.swing.JPanel;
015import javax.swing.JScrollPane;
016import javax.swing.JTextArea;
017import javax.swing.JTextField;
018import javax.swing.JButton;
019
020/**
021 * This class implements a simple graphical user interface with a text entry
022 * area, a text output area and an optional image.
023 * 
024 * @author Michael Kolling
025 * @version 1.0 (Jan 2003) DB edited (2019) CD edited (2022)
026 */
027public class UserInterface implements ActionListener {
028    private GameEngine aEngine;
029    private JFrame aMyFrame;
030    private JTextField aEntryField;
031    private JTextArea aLog;
032    private JLabel aImage;
033    private JButton aButton;
034
035    /**
036     * Construct a UserInterface. As a parameter, a Game Engine
037     * (an object processing and executing the game commands) is
038     * needed.
039     * 
040     * @param pGameEngine The GameEngine object implementing the game logic.
041     */
042    public UserInterface(final GameEngine pGameEngine) {
043        this.aEngine = pGameEngine;
044        this.createGUI();
045    } // UserInterface(.)
046
047    /**
048     * Print out some text into the text area.
049     * 
050     * @param pText like sysout but for gui, is the text in " "
051     */
052    public void print(final String pText) {
053        this.aLog.append(pText);
054        this.aLog.setCaretPosition(this.aLog.getDocument().getLength());
055    } // print(.)
056
057    /**
058     * Print out some text into the text area, followed by a line break.
059     * 
060     * @param pText like sysout but for gui, is the text in " "
061     */
062    public void println(final String pText) {
063        this.print(pText + "\n");
064    } // println(.)
065
066    /**
067     * Show an image file in the interface.
068     * 
069     * @param pImageName name of the image for the room
070     */
071    public void showImage(final String pImageName) {
072        String vImagePath = "" + pImageName; // to change the directory
073        URL vImageURL = this.getClass().getClassLoader().getResource(vImagePath);
074        if (vImageURL == null)
075            System.out.println("Image not found : " + vImagePath);
076        else {
077            ImageIcon vIcon = new ImageIcon(vImageURL);
078            this.aImage.setIcon(vIcon);
079            this.aMyFrame.pack();
080        }
081    } // showImage(.)
082
083    /**
084     * Enable or disable input in the input field.
085     * 
086     * @param pOnOff true when enable
087     *               false when disable
088     */
089    public void enable(final boolean pOnOff) {
090        this.aEntryField.setEditable(pOnOff); // enable/disable
091        if (!pOnOff) { // disable
092            this.aEntryField.getCaret().setBlinkRate(0); // cursor won't blink
093            this.aEntryField.removeActionListener(this); // won't react to entry
094        }
095    } // enable(.)
096
097    /**
098     * Set up graphical user interface.
099     */
100    private void createGUI() {
101        this.aMyFrame = new JFrame("Zuul GOTY Edition"); // change the title
102        this.aEntryField = new JTextField(34);
103        this.aButton = new JButton("quit");
104
105        this.aLog = new JTextArea();
106        this.aLog.setEditable(false);
107        JScrollPane vListScroller = new JScrollPane(this.aLog);
108        vListScroller.setPreferredSize(new Dimension(200, 200));
109        vListScroller.setMinimumSize(new Dimension(100, 100));
110
111        JPanel vPanel = new JPanel();
112        this.aImage = new JLabel();
113
114        vPanel.setLayout(new BorderLayout()); // ==> only five places
115        vPanel.add(this.aImage, BorderLayout.NORTH);
116        vPanel.add(vListScroller, BorderLayout.CENTER);
117        vPanel.add(this.aEntryField, BorderLayout.SOUTH);
118        vPanel.add(this.aButton, BorderLayout.WEST);
119
120        this.aMyFrame.getContentPane().add(vPanel, BorderLayout.CENTER);
121
122        // add some event listeners to some components
123        this.aEntryField.addActionListener(this);
124        this.aButton.addActionListener(this);
125
126        // to end program when window is closed
127        this.aMyFrame.addWindowListener(new WindowAdapter() {
128            public void windowClosing(WindowEvent e) {
129                System.exit(0);
130            }
131        });
132
133        this.aMyFrame.pack();
134        this.aMyFrame.setVisible(true);
135        this.aEntryField.requestFocus();
136    } // createGUI()
137
138    /**
139     * Actionlistener interface for entry textfield.
140     * 
141     * @param pE event of actionListener
142     */
143    public void actionPerformed(final ActionEvent pE) {
144        // check the type of action
145        if (pE.getActionCommand() != null) {
146            this.aEngine.interpretCommand(pE.getActionCommand());
147            this.aEntryField.setText(""); // to reset entry field
148        } else {
149            this.processCommand(); // never suppress this line
150        }
151    } // actionPerformed(.)
152
153    /**
154     * A command has been entered. Read the command and do whatever is
155     * necessary to process it.
156     */
157    private void processCommand() {
158        String vInput = this.aEntryField.getText();
159        this.aEntryField.setText("");
160
161        this.aEngine.interpretCommand(vInput);
162    } // processCommand()
163} // UserInterface

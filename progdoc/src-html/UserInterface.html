150    /**
151     * This method prints slowly the text in param
152     * 
153     * This code was found on internet and update for my use
154     * 
155     * @author https://github.com/Vourem/SlowPrint-Method/blob/master/SlowPrint
156     * @param pText Text you want to write slowly
157     */
158    public void slowPrint(final String pText) {
159
160        aTextTimer = new Timer(aTime, new ActionListener() {
161            @Override
162            public void actionPerformed(ActionEvent pEvent) {
163                aLog.setText(pText.substring(0, aIndex));
164                aIndex++;
165                if (aIndex > pText.length()) {
166                    ((Timer) pEvent.getSource()).stop();
167                }
168            }
169
170        });
171
172        aTextTimer.start();
173        aIndex = 0;
174    }

261    /**
262     * This method play sound.
263     * 
264     * This code was found on internet and update for my use
265     * 
266     * @author https://stackoverflow.com/questions/6045384/playing-mp3-and-wav-in-java
267     * @param pFile File you want to play
268     * @param pLoop Number of repetitions of the sound (-1 is infinite)
269     */
270    public void playSound(final String pFile, final int pLoop) {
271        try {
272            AudioInputStream vAudioInputStream = AudioSystem
273                    .getAudioInputStream(new File("gameSounds/" + pFile + ".wav").getAbsoluteFile());
274            aClip = AudioSystem.getClip();
275            aClip.open(vAudioInputStream);
276            aClip.start();
277            aClip.loop(pLoop);
278        } catch (Exception ex) {
279            System.out.println("Error with playing sound.");
280            ex.printStackTrace();
281        }
282    }

509    /**
510     * This method create custom font
511     */
512    private void createFont() {
513        // to import custom font
514        // https://www.ryisnow.online/2021/04/java-for-beginner-how-to-use-custom-font.html
515        try {
516            aFont = Font.createFont(Font.TRUETYPE_FONT, new File("font/8bit.ttf"));
517            GraphicsEnvironment vGraphicsEnvironment = GraphicsEnvironment.getLocalGraphicsEnvironment();
518            vGraphicsEnvironment.registerFont(aFont);
519
520            aTextFont = aFont.deriveFont(18f);
521            aButtonsFont = aFont.deriveFont(14f);
522            aMenuFont = aFont.deriveFont(25f);
523        } catch (IOException | FontFormatException e) {
524            e.printStackTrace();
525        }
526    }

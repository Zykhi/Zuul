087    /**
088     * This method prints slowly the text in param
089     * 
090     * This code was found on internet and update for my use
091     * 
092     * @author https://github.com/Vourem/SlowPrint-Method/blob/master/SlowPrint
093     * @param pText Text you want to write slowly
094     */
095    public void slowPrint(final String pText) {
096
097        for (char c : pText.toCharArray()) {
098            String vMessage = Character.toString(c);
099
100            this.aLog.append(vMessage);
101            this.aLog.setCaretPosition(this.aLog.getDocument().getLength());
102
103            try {
104                Thread.sleep(aTime);
105            } catch (InterruptedException e) {
106                e.printStackTrace();
107            }
108        }
109    }

112     * This method is like SlowPrint but for println
113     * 
114     * @param pText Text you want to write slowly
115     */
116    public void slowPrintln(final String pText) {
117        this.slowPrint(pText + "\n");
118    }

126     * @param pFile File you want to play

128    public void playSound(final String pFile) {
129        try {
130            AudioInputStream vAudioInputStream = AudioSystem
131                    .getAudioInputStream(new File("gameSounds/" + pFile + ".wav").getAbsoluteFile());
132            aClip = AudioSystem.getClip();
133            aClip.open(vAudioInputStream);
134            aClip.start();
135        } catch (Exception ex) {
136            System.out.println("Error with playing sound.");
137            ex.printStackTrace();
138        }
139    }

207    /**
208     * Actionlistener interface for entry textfield.
209     * 
210     * @param pE event of actionListener
211     */
212    public void actionPerformed(final ActionEvent pE) {
213        // check the type of action
214        if (pE.getActionCommand() != null) {
215            this.aEngine.interpretCommand(aParser.getCommand(pE.getActionCommand()));
216            this.aEntryField.setText(""); // to reset entry field
217        } else {
218            this.processCommand(); // never suppress this line
219        }
220    } // actionPerformed(.)

359    /**
360     * This method is called when you click on the drop button
361     * It's to change UI to show items in inventory
362     */
363    public void dropButtonMethod() {
364        JButton[] vButtons = { aBackButton, aHelpButton, aQuitButton, aDropButton, aTakeButton, aFireButton,
365                aChargeButton, aInventoryButton };
366
367        if (aEngine.getCurrentInventoryItemsString() != null) {
368            String[] vOutput = aEngine.getCurrentInventoryItemsString().split(" ");
369            for (int i = 0; i < vOutput.length; i++) {
370                vButtons[i].setText(vOutput[i]);
371                vButtons[i].setActionCommand("drop " + vOutput[i]);
372            }
373            for (int i = vOutput.length; i < 8; i++) {
374                vButtons[i].setText("");
375                vButtons[i].setActionCommand("");
376                vButtons[i].removeActionListener(this);
377            }
378            aSkipButton.setText("exit");
379            aSkipButton.setActionCommand("exit");
380            aSkipButton.addActionListener(e -> exitButtonMethod());
381            for (int i = vOutput.length; i < 8; i++) {
382                vButtons[i].addActionListener(this);
383            }
384        }
385    }

470    /**
471     * This method start the timer
472     * 
473     * @author https://www.developpez.net/forums/anocode.php?id=f5b514a1cac1a7a4fd52f16767094ab1
474     */
475    public void startTimer() {
476        aSecond = 0;
477        aMinute = 0;
478        aTaskTimer = new ActionListener() {
479            public void actionPerformed(ActionEvent pE) {
480
481                aSecond++;
482                if (aSecond == 60) {
483                    aSecond = 0;
484                    aMinute++;
485                }
486                aGameTimer.setText("elapsed time : " + aMinute + " : " + aSecond);
487            }
488        };
489
490        aTimer = new Timer(aDelay, aTaskTimer);
491        aTimer.start();
492    }
493

495     * this method stop the timer

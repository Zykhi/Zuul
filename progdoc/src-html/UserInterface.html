109    /**
110     * This method prints slowly the text in param
111     * 
112     * This code was found on internet and update for my use
113     * 
114     * @author https://github.com/Vourem/SlowPrint-Method/blob/master/SlowPrint
115     * @param pText Text you want to write slowly
116     */
117    public void slowPrint(final String pText) {
118
119        for (char c : pText.toCharArray()) {
120            String vMessage = Character.toString(c);
121
122            this.aLog.append(vMessage);
123            this.aLog.setCaretPosition(this.aLog.getDocument().getLength());
124
125            try {
126                Thread.sleep(aTime);
127            } catch (InterruptedException e) {
128                e.printStackTrace();
129            }
130        }
131    }

139        this.slowPrint(pText + "\n");

153    /**
154     * This method play sound.
155     * 
156     * This code was found on internet and update for my use
157     * 
158     * @author https://stackoverflow.com/questions/6045384/playing-mp3-and-wav-in-java
159     * @param pFile File you want to play
160     * @param pLoop Number of repetitions of the sound (-1 is infinite)
161     */
162    public void playSound(final String pFile, final int pLoop) {
163        try {
164            AudioInputStream vAudioInputStream = AudioSystem
165                    .getAudioInputStream(new File("gameSounds/" + pFile + ".wav").getAbsoluteFile());
166            aClip = AudioSystem.getClip();
167            aClip.open(vAudioInputStream);
168            aClip.start();
169            aClip.loop(pLoop);
170        } catch (Exception ex) {
171            System.out.println("Error with playing sound.");
172            ex.printStackTrace();
173        }
174    }

272            aFont = Font.createFont(Font.TRUETYPE_FONT, new File("font/8bit.ttf"));

471    /**
472     * This method is called when you click on the drop button
473     * It's to change UI to show items in inventory
474     */
475    public void dropButtonMethod() {
476        JButton[] vButtons = { aBackButton, aHelpButton, aQuitButton, aDropButton, aTakeButton, aFireButton,
477                aChargeButton, aInventoryButton };
478        if (aEngine.getCurrentInventoryItemsString() != null) {
479            String[] vOutput = aEngine.getCurrentInventoryItemsString().split(" ");
480            for (int i = 0; i < vOutput.length; i++) {
481                vButtons[i].setText(vOutput[i]);
482                vButtons[i].setActionCommand("drop " + vOutput[i]);
483            }
484            for (int i = vOutput.length; i < 8; i++) {
485                vButtons[i].setText("");
486                vButtons[i].setActionCommand("");
487                vButtons[i].removeActionListener(this);
488            }
489            aSkipButton.setText("exit");
490            aSkipButton.setActionCommand("exit");
491            aSkipButton.addActionListener(e -> exitButtonMethod());
492            for (int i = vOutput.length; i < 8; i++) {
493                vButtons[i].addActionListener(this);
494            }
495            this.println("What do you want to drop ?");
496        } else {
497            this.println("Your inventory is empty");
498        }
499    }

558    /**
559     * Actionlistener interface for entry textfield.
560     * 
561     * @param pE event of actionListener
562     */
563    public void actionPerformed(final ActionEvent pE) {
564        // check the type of action
565        if (pE.getActionCommand() != null) {
566            this.aEngine.interpretCommand(aParser.getCommand(pE.getActionCommand()));
567            this.aEntryField.setText(""); // to reset entry field
568        } else {
569            this.processCommand(); // never suppress this line
570        }
571    } // actionPerformed(.)
572
573    /**
574     * A command has been entered. Read the command and do whatever is
575     * necessary to process it.
576     */
577    private void processCommand() {
578        String vInput = this.aEntryField.getText();
579        this.aEntryField.setText("");
580
581        this.aEngine.interpretCommand(aParser.getCommand(vInput));
582    } // processCommand()

586    /**
587     * This method start the timer
588     * 
589     * @author https://www.developpez.net/forums/anocode.php?id=f5b514a1cac1a7a4fd52f16767094ab1
590     */
591    public void startTimer() {
592        aSecond = 0;
593        aMinute = 0;
594        aTaskTimer = new ActionListener() {
595            public void actionPerformed(ActionEvent pE) {
596
597                aSecond++;
598                if (aSecond == 60) {
599                    aSecond = 0;
600                    aMinute++;
601                }
602                aGameTimer.setText("elapsed time : " + aMinute + " : " + aSecond);
603            }
604        };
605
606        aTimer = new Timer(aDelay, aTaskTimer);
607        aTimer.start();
608    }

001/**
002 * Representations for all the valid command words for the game
003 * along with a string in a particular language.
004 * 
005 * @author Michael Kolling and David J. Barnes + C.Diouy
006 * @version 2006.03.30 + 2022.02.25
007 */
008public enum CommandWord {
009    // A value for each command word, plus one for unrecognised
010    // commands.
011    HELP("help"), GO("go"), QUIT("quit"), LOOK("look"), EAT("eat"), BACK("back"),
012    TEST("test"), TAKE("take"), DROP("drop"), INVENTORY("inventory"), UNKNOWN("?"),
013    CHARGE("charge"), FIRE("fire"), EXIT("exit"), SKIP("skip"), ALEA("alea");
014
015    // The command string.
016    private String aCommandString;
017
018    /**
019     * Initialise with the corresponding command word.
020     * 
021     * @param pCommandWord The command string.
022     */
023    CommandWord(String pCommandWord) {
024        this.aCommandString = pCommandWord;
025    }
026
027    /**
028     * @return The command word as a string.
029     */
030    public String toString() {
031        return aCommandString;
032    }
033}

001package pkg_Room;
002import java.util.HashMap;
003import java.util.ArrayList;
004import java.util.Random;
005
006/**
007 * Thid class create a random room
008 *
009 * @author C.Diouy
010 */
011public class RoomRandomizer {
012    private HashMap<String, Room> aRooms;
013
014    /**
015     * this constructor create a room randomizer
016     * 
017     * @param pRooms Hashmap contain all the room
018     */
019    public RoomRandomizer(final HashMap<String, Room> pRooms) {
020        this.aRooms = pRooms;
021    }
022
023    /**
024     * Find a random room
025     * 
026     * @return room in array list
027     */
028    public Room findRandomRoom() {
029        Random vRandomNb = new Random();
030        ArrayList<Room> vRoomsList = new ArrayList<Room>(this.aRooms.values());
031        int vNb = vRandomNb.nextInt(vRoomsList.size());
032        return vRoomsList.get(vNb);
033    }
034}

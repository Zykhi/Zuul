001/**
002 * This class is part of the "Zuul GOTY Edition" application.
003 * "Zuul GOTY Edition" is a very simple, text based adventure game.
004 * 
005 * This class holds an enumeration table of all command words known to the game.
006 * It is used to recognise commands as they are typed in.
007 *
008 * @author Michael Kolling and David J. Barnes + D.Bureau
009 * @version 2008.03.30 + 2019.09.25
010 */
011public class CommandWords {
012    // a constant array that will hold all valid command words
013    private final String[] aValidCommands;
014
015    /**
016     * Constructor - initialise the command words.
017     */
018    public CommandWords() {
019        this.aValidCommands = new String[10];
020        this.aValidCommands[0] = "go";
021        this.aValidCommands[1] = "help";
022        this.aValidCommands[2] = "quit";
023        this.aValidCommands[3] = "look";
024        this.aValidCommands[4] = "eat";
025        this.aValidCommands[5] = "back";
026        this.aValidCommands[6] = "test";
027        this.aValidCommands[7] = "take";
028        this.aValidCommands[8] = "drop";
029        this.aValidCommands[9] = "inventory";
030    } // CommandWords()
031
032    /**
033     * Check whether a given String is a valid command word.
034     * 
035     * @param pString Command line
036     * 
037     * @return true if a given string is a valid command,
038     *         false if it isn't.
039     */
040    public boolean isCommand(final String pString) {
041        for (int i = 0; i < this.aValidCommands.length; i++) {
042            if (this.aValidCommands[i].equals(pString))
043                return true;
044        } // for
045          // if we get here, the string was not found in the commands
046        return false;
047    } // isCommand()
048
049    /**
050     * @return a String of all valid commands.
051     */
052    public String getCommandList() {
053        StringBuilder sCommands = new StringBuilder();
054        for (int i = 0; i < aValidCommands.length; i++) {
055            sCommands.append(aValidCommands[i] + "  ");
056        }
057        return sCommands.toString();
058    }
059} // CommandWords

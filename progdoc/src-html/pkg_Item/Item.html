001package pkg_Item;
002
003/**
004 * This class implement item for the game
005 * 
006 * @author C.Diouy
007 * @version 2022.02.19
008 */
009public class Item {
010    private String aName;
011    private double aPrice;
012    private double aWeight;
013    private String aDescription;
014    private boolean aFightableItem;
015
016    /**
017     * Create an item with a name, a price, a weight and a description
018     * 
019     * @param pName        name of the item
020     * @param pPrice       price of the item
021     * @param pWeight      weight of the item
022     * @param pDescription description of the item
023     * @param pFightableItem   true if the item is fightable
024     */
025    public Item(final String pName, final double pPrice, final double pWeight, final String pDescription,
026            boolean pFightableItem) {
027        this.aName = pName;
028        this.aPrice = pPrice;
029        this.aWeight = pWeight;
030        this.aDescription = pDescription;
031        this.aFightableItem = pFightableItem;
032    }
033
034    /**
035     * Redefine toString()
036     * 
037     * @return a string with all informations about an item
038     */
039    @Override
040    public String toString() {
041        return this.aName +
042                ", price : " + this.aPrice +
043                ", weight : " + this.aWeight + "kg, " +
044                this.aDescription;
045    }
046
047    /**
048     * This String get a description
049     * 
050     * @return the description of the item (the one that was defined in the
051     *         constructor).
052     */
053    public String getDescription() {
054        return this.aDescription;
055    }
056
057    /**
058     * This String get the name of the item
059     * 
060     * @return the name of the item (the one that was defined in the
061     *         constructor).
062     */
063    public String getName() {
064        return this.aName;
065    }
066
067    /**
068     * This double get the price of the item
069     * 
070     * @return the price of the item (the one that was defined in the
071     *         constructor).
072     */
073    public double getPrice() {
074        return this.aPrice;
075    }
076
077    /**
078     * This double get the name of the item
079     * 
080     * @return the weight of the item (the one that was defined in the
081     *         constructor).
082     */
083    public double getWeight() {
084        return this.aWeight;
085    }
086
087    /**
088     * This function get if the item is a fightable item
089     * 
090     * @return true if the item is a fightable item, false otherwise
091     */
092    public boolean isFightableItem() {
093        return this.aFightableItem;
094    }
095
096}

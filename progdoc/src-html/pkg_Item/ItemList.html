001package pkg_Item;
002
003import java.util.HashMap;
004import java.util.Set;
005
006/**
007 * Item list class -
008 * 
009 * @author C.Diouy
010 * @version 2022.02.22
011 */
012public class ItemList {
013    private HashMap<String, Item> aItems;
014    private double aWeight;
015
016    /**
017     * Constructor of item list
018     */
019    public ItemList() {
020        this.aItems = new HashMap<String, Item>();
021        this.aWeight = 0;
022    }
023
024    /**
025     * This function get the name of the item
026     * 
027     * @param pName name of the item
028     * @return a item
029     */
030    public Item getItemName(final String pName) {
031        return this.aItems.get(pName);
032    }
033
034    /**
035     * This function get the weight of the item
036     * 
037     * @return a double with the weight of the item
038     */
039    public double getWeight() {
040        return this.aWeight;
041    }
042
043    /**
044     * This String get all the item in a String
045     * 
046     * @return a string describing the room's items, for example
047     *         "Items : sword shield"
048     */
049    public String getItemString() {
050        String vItemString = "Items in the room :";
051        Set<String> vKeys = aItems.keySet();
052        for (String vItem : vKeys) {
053            vItemString += " " + vItem;
054        }
055        return vItemString;
056    }
057
058    /**
059     * This function get all the item in inventory in a string
060     * 
061     * @return a string describing the inventory's items
062     */
063    public String getInventoryString() {
064        if (this.aItems.isEmpty()) {
065            return "Your inventory is empty";
066        } else {
067            StringBuilder vInventoryBuilder = new StringBuilder("Your inventory : ");
068            Set<String> vKeys = aItems.keySet();
069            for (String vS : vKeys) {
070                vInventoryBuilder.append(" " + vS);
071            }
072            return vInventoryBuilder.toString();
073        }
074    }
075
076    /**
077     * This function get all the fightable item in inventory in a string
078     * 
079     * @return a string describing the inventory's fightable items
080     */
081    public String getInventoryFightableString() {
082        if (this.aItems.isEmpty()) {
083            return "Your inventory is empty";
084        } else {
085            StringBuilder vInventoryBuilder = new StringBuilder("Your inventory : ");
086            Set<String> vKeys = aItems.keySet();
087            for (String vS : vKeys) {
088                if (aItems.get(vS).isFightableItem() == true) {
089                    vInventoryBuilder.append(" " + vS);
090                }
091            }
092            return vInventoryBuilder.toString();
093        }
094    }
095
096    /**
097     * This function get the inventory
098     * 
099     * @return inventory
100     */
101    public HashMap<String, Item> getItemList() {
102        return this.aItems;
103    }
104
105    /**
106     * This method add an item
107     * 
108     * @param pName name of the item
109     * @param pItem the item
110     */
111    public void addItem(final String pName, final Item pItem) {
112        this.aItems.put(pName, pItem);
113    }
114
115    /**
116     * This method remove an item
117     * 
118     * @param pName name of the item
119     * @param pItem the item
120     */
121    public void removeItem(final String pName, final Item pItem) {
122        this.aItems.remove(pName, pItem);
123    }
124
125    /**
126     * This method add the weight of item
127     * 
128     * @param pItemWeight the weight of the item
129     */
130    public void addWeight(final double pItemWeight) {
131        this.aWeight += pItemWeight;
132    }
133
134    /**
135     * This method remove the weight of item
136     * 
137     * @param pItemWeight the weight of the item
138     */
139    public void removeWeight(final double pItemWeight) {
140        this.aWeight -= pItemWeight;
141    }
142
143    /**
144     * This function get the String of the weight of inventory
145     * 
146     * @return the string of the weight of the inventory
147     */
148    public String getWeightString() {
149        return "Weight : " + this.getWeight() + "kg / ";
150    }
151
152    /**
153     * Check if contain a specific item
154     * 
155     * @param pItem The specific item
156     * @return true if it's the specific item
157     *         false if it's not
158     */
159    public boolean contain(final Item pItem) {
160        return this.aItems.containsValue(pItem);
161    }
162
163    /**
164     * Check if inventory is empty
165     * 
166     * @return true if inventory is empty
167     *         false if there is item inside
168     */
169    public boolean isEmpty() {
170        return this.aItems.isEmpty();
171    }
172}

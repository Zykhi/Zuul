001package pkg_Item;
002import pkg_Room.Room;
003
004/**
005 * This class implements beamer
006 *
007 * @author C.Diouy
008 * @version 2022.02.26
009 */
010public class Beamer extends Item {
011    private boolean aIsCharged;
012    private Room aRoomCharged;
013
014    /**
015     * Create beamer
016     */
017    public Beamer() {
018        super("teleporter", 10, 5, "teleport where it's charged", false);
019        this.aIsCharged = false;
020        this.aRoomCharged = null;
021    }
022
023    /**
024     * charge the beamer
025     * 
026     * @param pRoom room where you charge the beamer
027     */
028    public void charge(final Room pRoom) {
029        this.aIsCharged = true;
030        this.aRoomCharged = pRoom;
031    }
032
033    /**
034     * for teleport
035     * 
036     * @return at the room where you charge the beamer
037     */
038    public Room fire() {
039        this.aIsCharged = false;
040        Room vRoom = this.aRoomCharged;
041        this.aRoomCharged = null;
042        return vRoom;
043    }
044
045    /**
046     * @return true if is charged
047     *         false if isnt
048     */
049    public boolean isCharged() {
050        return this.aIsCharged;
051    }
052}

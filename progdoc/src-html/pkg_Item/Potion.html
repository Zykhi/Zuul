001package pkg_Item;
002
003/**
004 * This class implement potion item
005 */
006public class Potion extends Item {
007    private int aHealing;
008
009    /**
010     * Create a potion whith a healing value
011     * 
012     * @param pHealingPoint healing point of the potion
013     */
014    public Potion(int pHealingPoint) {
015        super("potion", 0, 0, "this is a potion who heal " + pHealingPoint + " hp", true);
016        aHealing = pHealingPoint;
017    }
018
019    /**
020     * This function get the healing point of the potion
021     * 
022     * @return the healing point of the potion
023     */
024    public int getHealingPoint() {
025        return this.aHealing;
026    }
027
028}

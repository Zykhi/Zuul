006public class Door {
007    private boolean aIsTrap;
008
009    /**
010     * This constructor create door
011     * 
012     * @param pTrap true if door is trap
013     *              false if it's isnt
014     */
015    public Door(final boolean pTrap) {
016        this.aIsTrap = pTrap;
017    }
018
019    /**
020     * get if is trapdoor
021     * 
022     * @return true if door is trap
023     *         false if it's isnt
024     */
025    public boolean isTrap() {
026        return this.aIsTrap;
027    }
028}

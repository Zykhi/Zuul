001/**

006 */
007public class Door {
008    private boolean aIsTrap;
009
010    /**
011     * This constructor create door
012     * 
013     * @param pTrap true if door is trap
014     *              false if it's isnt
015     */
016    public Door(final boolean pTrap) {
017        this.aIsTrap = pTrap;
018    }
019
020    /**
021     * get if is trapdoor
022     * 
023     * @return true if door is trap
024     *         false if it's isnt
025     */
026    public boolean isTrap() {
027        return this.aIsTrap;
028    }
029}

001import java.util.HashMap;
002import java.util.Set;
003
004/**
005 * Item list class -
006 * 
007 * @author C.Diouy
008 * @version 2022.02.22
009 */
010public class ItemList {
011    private HashMap<String, Item> aItems;
012    private double aWeight;
013
014    /**
015     * Constructor of item list
016     */
017    public ItemList() {
018        this.aItems = new HashMap<String, Item>();
019        this.aWeight = 0;
020    }
021
022    /**
023     * This function get the name of the item
024     * 
025     * @param pName name of the item
026     * @return a item
027     */
028    public Item getItemName(final String pName) {
029        return this.aItems.get(pName);
030    }
031
032    /**
033     * This function get the weight of the item
034     * 
035     * @return a double with the weight of the item
036     */
037    public double getWeight() {
038        return this.aWeight;
039    }
040
041    /**
042     * This String get all the item in a String
043     * 
044     * @return a string describing the room's items, for example
045     *         "Items : sword shield"
046     */
047    public String getItemString() {
048        String vItemString = "Items in the room :";
049        Set<String> vKeys = aItems.keySet();
050        for (String vItem : vKeys) {
051            vItemString += " " + vItem;
052        }
053        return vItemString;
054    }
055
056    /**
057     * This function get all the item in inventory in a string
058     * 
059     * @return a string describing the inventory's items
060     */
061    public String getInventoryString() {
062        if (this.aItems.isEmpty()) {
063            return "Your inventory is empty";
064        } else {
065            StringBuilder vInventoryBuilder = new StringBuilder("Your inventory : ");
066            Set<String> vKeys = aItems.keySet();
067            for (String vS : vKeys) {
068                vInventoryBuilder.append(" " + vS);
069            }
070            return vInventoryBuilder.toString();
071        }
072    }
073
074    /**
075     * This function get the inventory
076     * 
077     * @return inventory
078     */
079    public HashMap<String, Item> getItemList() {
080        return this.aItems;
081    }
082
083    /**
084     * This method add an item
085     * 
086     * @param pName name of the item
087     * @param pItem the item
088     */
089    public void addItem(final String pName, final Item pItem) {
090        this.aItems.put(pName, pItem);
091    }
092
093    /**
094     * This method remove an item
095     * 
096     * @param pName name of the item
097     * @param pItem the item
098     */
099    public void removeItem(final String pName, final Item pItem) {
100        this.aItems.remove(pName, pItem);
101    }
102
103    /**
104     * This method add the weight of item
105     * 
106     * @param pItemWeight the weight of the item
107     */
108    public void addWeight(final double pItemWeight) {
109        this.aWeight += pItemWeight;
110    }
111
112    /**
113     * This method remove the weight of item
114     * 
115     * @param pItemWeight the weight of the item
116     */
117    public void removeWeight(final double pItemWeight) {
118        this.aWeight -= pItemWeight;
119    }
120
121    /**
122     * This function get the String of the weight of inventory
123     * 
124     * @return the string of the weight of the inventory
125     */
126    public String getWeightString() {
127        return "Weight : " + this.getWeight() + "kg / ";
128    }
129
130    /**
131     * Check if contain a specific item
132     * 
133     * @param pItem The specific item
134     * @return true if it's the specific item
135     *         false if it's not
136     */
137    public boolean contain(final Item pItem) {
138        return this.aItems.containsValue(pItem);
139    }
140
141    /**
142     * Check if inventory is empty
143     * 
144     * @return true if inventory is empty
145     *         false if there is item inside
146     */
147    public boolean isEmpty() {
148        return this.aItems.isEmpty();
149    }
150}

001/**
002 * This class implement item for the game
003 * 
004 * @author C.Diouy
005 * @version 2022.02.19
006 */
007public class Item {
008    private String aName;
009    private double aPrice;
010    private double aWeight;
011    private String aDescription;
012
013    /**
014     * Create an item with a name, a price, a weight and a description
015     * 
016     * @param pName        name of the item
017     * @param pPrice       price of the item
018     * @param pWeight      weight of the item
019     * @param pDescription description of the item
020     */
021    public Item(final String pName, final double pPrice, final double pWeight, final String pDescription) {
022        this.aName = pName;
023        this.aPrice = pPrice;
024        this.aWeight = pWeight;
025        this.aDescription = pDescription;
026    }
027
028    /**
029     * Redefine toString()
030     * 
031     * @return a string with all informations about an item
032     */
033    @Override
034    public String toString() {
035        return this.aName +
036                ", price : " + this.aPrice +
037                ", weight : " + this.aWeight + "kg, " +
038                this.aDescription;
039    }
040
041    /**
042     * This String get a description
043     * 
044     * @return the description of the item (the one that was defined in the
045     *         constructor).
046     */
047    public String getDescription() {
048        return this.aDescription;
049    }
050
051    /**
052     * This String get the name of the item
053     * 
054     * @return the name of the item (the one that was defined in the
055     *         constructor).
056     */
057    public String getName() {
058        return this.aName;
059    }
060
061    /**
062     * This double get the price of the item
063     * 
064     * @return the price of the item (the one that was defined in the
065     *         constructor).
066     */
067    public double getPrice() {
068        return this.aPrice;
069    }
070
071    /**
072     * This double get the name of the item
073     * 
074     * @return the weight of the item (the one that was defined in the
075     *         constructor).
076     */
077    public double getWeight() {
078        return this.aWeight;
079    }
080
081}

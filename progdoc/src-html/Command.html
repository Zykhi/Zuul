001/**
002 * This class is part of the "Zuul GOTY Edition" application.
003 * "Zuul GOTY Edition" is a very simple, text based adventure game.
004 * 
005 * This class holds information about a command that was issued by the user.
006 * A command currently consists of two strings: a command word and a second
007 * word (for example, if the command was "take map", then the two strings
008 * obviously are "take" and "map").
009 * 
010 * The way this is used is: Commands are already checked for being valid
011 * command words. If the user entered an invalid command (a word that is not
012 * known) then the command word is null.
013 *
014 * If the command had only one word, then the second word is null.
015 *
016 * @author Michael Kolling and David J. Barnes + D.Bureau + C.Diouy
017 * @version 2008.03.30 + 2019.09.25 + 2022.02.11
018 */
019public class Command {
020    private CommandWord aCommandWord;
021    private String aSecondWord;
022
023    /**
024     * Create a command object. First and second word must be supplied, but
025     * either one (or both) can be null. The command word should be null to
026     * indicate that this was a command that is not recognised by this game.
027     * 
028     * @param pCommandWord first word like "go" or "help"
029     * @param pSecondWord  second word if it's needed like "north"
030     */
031    public Command(final CommandWord pCommandWord, final String pSecondWord) {
032        this.aCommandWord = pCommandWord;
033        this.aSecondWord = pSecondWord != null ? pSecondWord.toLowerCase() : null;
034    }
035
036    /**
037     * This String get the command word
038     * 
039     * @return the command word (the first word) of this command. If the
040     *         command was not understood, the result is null.
041     */
042    public CommandWord getCommandWord() {
043        return this.aCommandWord;
044    }
045
046    /**
047     * This String get the second word
048     * 
049     * @return the second word of this command. Returns null if there was no
050     *         second word.
051     */
052    public String getSecondWord() {
053        return this.aSecondWord;
054    }
055
056    /**
057     * This boolean check if there is a second word
058     * 
059     * @return true if the command has a second word.
060     */
061    public boolean hasSecondWord() {
062        return this.aSecondWord != null;
063    }
064
065    /**
066     * This boolean check if the command is known
067     * 
068     * @return true if this command was not understood.
069     */
070    public boolean isUnknown() {
071        return this.aCommandWord == CommandWord.UNKNOWN;
072    }
073} // Command

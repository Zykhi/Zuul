089    /**
090     * Print out some help information.
091     * Here we print some stupid, cryptic message and a list of the
092     * command words.
093     */
094    protected void printHelp() {
095        this.aGui.println("You are lost. You leave the fight.");
096        this.aGui.println("You wander around the dungeon.");
097        this.aGui.println("");
098        this.aGui.println(aParser.getCommandString());
099    }

101    /**
102     * Try to go to one direction. If there is an exit, enter the new
103     * room, otherwise print an error message.
104     * 
105     * @param pDirection use for check if there are second word
106     */
107    protected void goRoom(final Command pDirection) {
108        if (!pDirection.hasSecondWord()) {
109            // if there is no second word, we don't know where to go...
110            this.aGui.println("Go where?");
111            return;
112        }
113        this.getPreviousRooms().push(this.getCurrentRoom());
114
115        String vDirection = pDirection.getSecondWord();
116        Door vDoor = this.getCurrentRoom().getDoor(vDirection);
117
118        // Try to leave current room.
119        Room vNextRoom = this.getCurrentRoom().getExit(vDirection);
120
121        if (vNextRoom == null) {
122            this.aGui.println("There is no door!");
123        } else if (vDoor != null) {
124            if (vDoor.isTrap()) {
125                this.aGui.println("its a trap");
126                clearStack();
127                return;
128            }
129        } else {
130            this.setRoom(vNextRoom);
131            printLocationInfo();
132            aMovement -= 1;
133
134        }
135    }
136
137    /**
138     * This method print the info of the room with the exit when you wrote in a chat
139     * when there isnt second word, if there is it's print information about item
140     * 
141     * @param pCommand to check if there is second word to change between look and
142     *                 look item
143     */
144    protected void look(final Command pCommand) {
145        if (pCommand.hasSecondWord()) {
146            this.lookItem(pCommand);
147        } else {
148            printLocationInfo();
149        }
150    }
151
152    /**
153     * This method print the description of the item passed as a param
154     * 
155     * @param pItemName to get the item name
156     */
157    protected void lookItem(final Command pItemName) {
158        String vItemName = pItemName.getSecondWord();
159
160        Item vItem = this.getCurrentRoom().getItemName(vItemName);
161
162        if (vItem == null) {
163            this.aGui.println("I dont know what do you mean");
164        } else {
165            this.aGui.println(vItem.toString());
166        }
167    }

169    /**
170     * This method is to eat a cookie (for the moment)
171     * 
172     * @param pCommand to be sure there is a second word
173     */
174    protected void eat(final Command pCommand) {
175        String vItemName = pCommand.getSecondWord();
176        Item vItem = this.aInventory.getItemName(vItemName);
177        if (!pCommand.hasSecondWord()) {
178            this.aGui.println("what do you want to eat");
179        } else {
180            if (vItemName.equals("Cookie") && this.aInventory.getItemList().contain(vItem)) {
181                this.aMaxWeight *= 2;
182                this.aInventory.removeItem(vItemName, vItem);
183                this.aInventory.removeWeight(vItem.getWeight());
184                this.aGui.println("You eat a cookie");
185                showInventory();
186            } else {
187                this.aGui.println("You cant eat that");
188            }
189        }
190    }
191
192    /**
193     * This method is to return to the previous room to the starting room
194     * 
195     * @param pCommand to be sure there is no second word
196     */
197    protected void back(final Command pCommand) {
198        if (pCommand.hasSecondWord()) {
199            this.aGui.println("it's impossible");
200        } else if (this.getPreviousRooms().empty()) { // was aPreviousRooms.empty()
201            this.aGui.println("you cant back");
202        } else {
203            Room vPreviousRoom = this.getPreviousRooms().pop(); // was aPreviousRooms.pop()
204            this.setRoom(vPreviousRoom); // was aCurrentRoom = vPreviousRoom
205            printLocationInfo();
206            aMovement -= 1;
207        }
208    }
209
210    /**
211     * This method is to take an item
212     * 
213     * @param pItemName name of the item to take
214     */
215    protected void take(final Command pItemName) {
216        String vItemName = pItemName.getSecondWord();
217        Item vItem = this.aCurrentRoom.getItemName(vItemName);
218        if (!pItemName.hasSecondWord()) {
219            this.aGui.println("What do you want to take");
220        } else if (vItem == null) {
221            this.aGui.println("This is not here");
222        } else if (this.aInventory.getWeight() + vItem.getWeight() > this.aMaxWeight) {
223            this.aGui.println("Your inventory is full. Drop some items");
224        } else {
225
226            this.aInventory.addItem(vItemName, vItem);
227            this.aInventory.addWeight(vItem.getWeight());
228            this.aCurrentRoom.removeItem(vItemName, vItem);
229            showInventory();
230            this.aGui.println(this.aCurrentRoom.getItemString());
231        }
232    }
233
234    /**
235     * This method is to drop an item
236     * 
237     * @param pItemName name of the item to drop
238     */
239    protected void drop(final Command pItemName) {
240        String vItemName = pItemName.getSecondWord();
241        Item vItem = this.aInventory.getItemName(vItemName);
242        if (!pItemName.hasSecondWord()) {
243            this.aGui.println("What do you want to drop");
244        } else if (vItem == null) {
245            this.aGui.println("You dont have this");
246        } else {
247
248            this.aInventory.removeItem(vItemName, vItem);
249            this.aInventory.removeWeight(vItem.getWeight());
250            this.aCurrentRoom.addItem(vItemName, vItem);
251            showInventory();
252            this.aGui.println(this.aCurrentRoom.getItemString());
253        }
254    }

256    /**
257     * This method print all info about inventory of the player
258     */
259    protected void showInventory() {
260        this.aGui.println(this.aInventory.getInventoryString());
261        this.aGui.println(this.aInventory.getWeightString() + this.aMaxWeight + "kg");
262    }

264    /**
265     * This method charge the beamer
266     */
267    public void charge() {
268        Beamer vBeamer = (Beamer) this.aInventory.getItemName("teleporter");
269        if (this.aInventory.contain(vBeamer)) {
270            vBeamer.charge(this.aCurrentRoom);
271            this.aGui.println("Teleporter is charged");
272        } else {
273            this.aGui.println("You dont have teleporter");
274        }
275    }
276
277    /**
278     * This method fire the beamer
279     */
280    public void fire() {
281        Beamer vBeamer = (Beamer) this.aInventory.getItemName("teleporter");
282        if (this.aInventory.contain(vBeamer)) {
283            if (vBeamer.isCharged()) {
284                this.setRoom(vBeamer.fire());
285                this.aInventory.removeItem("teleporter", vBeamer);
286                this.aGui.println("You have been teleported. Teleporter is destroyed");
287                printLocationInfo();
288                showInventory();
289            } else {
290                this.aGui.println("teleporter isnt loaded.");
291            }
292        } else {
293            this.aGui.println("You dont have teleporter.");
294        }
295    }
296
297    /**
298     * This method print the info of the room with the exit when you enter on it
299     */
300    protected void printLocationInfo() {
301        this.aGui.println(this.getCurrentRoom().getLongDescription());
302        if (this.getCurrentRoom().getImageName() != null) {
303            this.aGui.showImage(this.getCurrentRoom().getImageName());
304        }
305    }

308     * This method clear the stack
309     */
310    public void clearStack() {

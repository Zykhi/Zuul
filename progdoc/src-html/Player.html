131    /**
132     * Try to go to one direction. If there is an exit, enter the new
133     * room, otherwise print an error message.
134     * 
135     * @param pDirection use for check if there are second word
136     */
137    protected void goRoom(final Command pDirection) {
138        if (!pDirection.hasSecondWord()) {
139            // if there is no second word, we don't know where to go...
140            this.aGui.println("Go where?");
141            return;
142        }
143        this.getPreviousRooms().push(this.getCurrentRoom());
144
145        String vDirection = pDirection.getSecondWord();
146        Door vDoor = this.getCurrentRoom().getDoor(vDirection);
147
148        // Try to leave current room.
149        Room vNextRoom = this.getCurrentRoom().getExit(vDirection);
150        if (vNextRoom == null) {
151            this.aGui.println("There is no door!");
152        } else if (vDoor != null) {
153            if (vDoor.isTrap()) {
154                this.aGui.println("its a trap");
155                clearStack();
156                return;
157            }
158        } else {
159            changeRoom(vNextRoom);
160        }
161    }

163    /**
164     * This method is called when the player change the current room
165     * 
166     * @param vNextRoom the next room
167     */
168    private void changeRoom(Room vNextRoom) {
169        this.setRoom(vNextRoom);
170        this.aGui.println("");
171        showImage();
172        printLocationInfo();
173        if (aCurrentRoom.getCharacter() != null) {
174            this.aGui.clearDialogArea();
175            this.aGui.showCharacterPanel();
176            showCharacter();
177            printCharacterDialog();
178            // this.aGui.playDialogSound(this.aCurrentRoom.getCharacterName());
179        } else {
180            this.aGui.hideCharacterPanel();
181        }
182        this.aCurrentRoom.updateNbrRoom();
183        this.aGui.playRoomSound();
184        aMovement -= 1;
185
186        if (this.getMovement() == 40) {
187            int vDelay = 100;// specify the delay for the timer
188            Timer vTimer = new Timer(vDelay, e -> {
189                // The following code will be executed once the delay is reached
190                this.printWelcome();
191            });
192            vTimer.setRepeats(false);// make sure the timer only runs once
193            vTimer.start();
194
195            int vDelay2 = 26000;// specify the delay for the timer
196            Timer vTimer2 = new Timer(vDelay2, e -> {
197                // The following code will be executed once the delay is reached
198                this.aGui.println("Type '" + CommandWord.HELP.toString() + "' if you need help." + '\n'
199                        + "You have 20 minutes to escape from the dungeon before being trapped forever" + '\n');
200                this.printLocationInfo();
201                this.aGui.startTimer();
202            });
203            vTimer2.setRepeats(false);// make sure the timer only runs once
204            vTimer2.start();
205        }
206    }

241    /**
242     * This method is to eat a cookie (for the moment)
243     * 
244     * @param pCommand to be sure there is a second word
245     */
246    protected void eat(final Command pCommand) {
247        String vItemName = pCommand.getSecondWord();
248        Item vItem = this.aInventory.getItemName(vItemName);
249        if (!pCommand.hasSecondWord()) {
250            this.aGui.println("what do you want to eat");
251        } else {
252            if (vItemName.equals("Cookie") && this.aInventory.getItemList().containsValue(vItem)) {
253                this.aMaxWeight *= 2;
254                this.aInventory.removeItem(vItemName, vItem);
255                this.aInventory.removeWeight(vItem.getWeight());
256                this.aGui.println("You eat a cookie");
257                showInventory();
258            } else {
259                this.aGui.println("You cant eat that");
260            }
261        }
262    }

347    /**
348     * This method fire the beamer
349     */
350    public void fire() {
351        Beamer vBeamer = (Beamer) this.aInventory.getItemName("teleporter");
352        if (this.aInventory.contain(vBeamer)) {
353            if (vBeamer.isCharged()) {
354                this.setRoom(vBeamer.fire());
355                this.aInventory.removeItem("teleporter", vBeamer);
356                this.aGui.println("You have been teleported. Teleporter is destroyed");
357                printLocationInfo();
358                showInventory();
359            } else {
360                this.aGui.println("teleporter isnt loaded.");
361            }
362        } else {
363            this.aGui.println("You dont have teleporter.");
364        }
365    }

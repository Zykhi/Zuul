137    /**
138     * Try to go to one direction. If there is an exit, enter the new
139     * room, otherwise print an error message.
140     * 
141     * @param pDirection use for check if there are second word
142     */
143    protected void goRoom(final Command pDirection) {
144        if (!pDirection.hasSecondWord()) {
145            // if there is no second word, we don't know where to go...
146            this.aGui.println("Go where?");
147            return;
148        }
149        this.getPreviousRooms().push(this.getCurrentRoom());
150
151        String vDirection = pDirection.getSecondWord();
152        Door vDoor = this.getCurrentRoom().getDoor(vDirection);
153
154        // Try to leave current room.
155        Room vNextRoom = this.getCurrentRoom().getExit(vDirection);
156        if (vNextRoom == null) {
157            this.aGui.println("There is no door!");
158        } else if (vDoor != null) {
159            if (vDoor.isTrap()) {
160                this.aGui.println("its a trap");
161                clearStack();
162                return;
163            }
164        } else {
165            changeRoom(vNextRoom);
166        }
167    }

169    /**
170     * This method is called when the player change the current room

172     */
173    private void changeRoom(Room vNextRoom) {
174        this.setRoom(vNextRoom);
175        showImage();
176        printLocationInfo();
177        this.aGui.playRoomSound();
178        aMovement -= 1;
179    }

214    /**
215     * This method is to eat a cookie (for the moment)
216     * 
217     * @param pCommand to be sure there is a second word
218     */
219    protected void eat(final Command pCommand) {
220        String vItemName = pCommand.getSecondWord();
221        Item vItem = this.aInventory.getItemName(vItemName);
222        if (!pCommand.hasSecondWord()) {
223            this.aGui.println("what do you want to eat");
224        } else {
225            if (vItemName.equals("Cookie") && this.aInventory.getItemList().containsValue(vItem)) {
226                this.aMaxWeight *= 2;
227                this.aInventory.removeItem(vItemName, vItem);
228                this.aInventory.removeWeight(vItem.getWeight());
229                this.aGui.println("You eat a cookie");
230                showInventory();
231            } else {
232                this.aGui.println("You cant eat that");
233            }
234        }
235    }

237    /**
238     * This method is to return to the previous room to the starting room
239     * 
240     * @param pCommand to be sure there is no second word
241     */
242    protected void back(final Command pCommand) {
243        if (pCommand.hasSecondWord()) {
244            this.aGui.println("it's impossible");
245        } else if (this.getPreviousRooms().empty()) { // was aPreviousRooms.empty()
246            this.aGui.println("you cant back");
247        } else {
248            Room vPreviousRoom = this.getPreviousRooms().pop(); // was aPreviousRooms.pop()
249            changeRoom(vPreviousRoom);
250        }
251    }

256     * @param pItemName name of the item to take

259        String vItemName = pItemName.getSecondWord();
260        Item vItem = this.aCurrentRoom.getItemName(vItemName);
261        if (!pItemName.hasSecondWord()) {
262
263        } else if (vItem == null) {
264            this.aGui.println("This is not here");
265        } else if (this.aInventory.getWeight() + vItem.getWeight() > this.aMaxWeight) {
266            this.aGui.println("Your inventory is full. Drop some items");
267        } else {
268
269            this.aInventory.addItem(vItemName, vItem);
270            this.aInventory.addWeight(vItem.getWeight());
271            this.aCurrentRoom.removeItem(vItemName, vItem);
272            showInventory();
273            this.aGui.println(this.aCurrentRoom.getItemString());
274        }

307    /**
308     * This method charge the beamer
309     */
310    public void charge() {
311        Beamer vBeamer = (Beamer) this.aInventory.getItemName("teleporter");
312        if (this.aInventory.contain(vBeamer)) {
313            vBeamer.charge(this.aCurrentRoom);
314            this.aGui.println("Teleporter is charged");
315        } else {
316            this.aGui.println("You dont have teleporter");
317        }
318    }

320    /**
321     * This method fire the beamer
322     */
323    public void fire() {
324        Beamer vBeamer = (Beamer) this.aInventory.getItemName("teleporter");
325        if (this.aInventory.contain(vBeamer)) {
326            if (vBeamer.isCharged()) {
327                this.setRoom(vBeamer.fire());
328                this.aInventory.removeItem("teleporter", vBeamer);
329                this.aGui.println("You have been teleported. Teleporter is destroyed");
330                printLocationInfo();
331                showInventory();
332            } else {
333                this.aGui.println("teleporter isnt loaded.");
334            }
335        } else {
336            this.aGui.println("You dont have teleporter.");
337        }
338    }

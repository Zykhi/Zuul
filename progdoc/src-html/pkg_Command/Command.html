001package pkg_Command;
002/**
003 * This class is part of the "Zuul GOTY Edition" application.
004 * "Zuul GOTY Edition" is a very simple, text based adventure game.
005 * 
006 * This class holds information about a command that was issued by the user.
007 * A command currently consists of two strings: a command word and a second
008 * word (for example, if the command was "take map", then the two strings
009 * obviously are "take" and "map").
010 * 
011 * The way this is used is: Commands are already checked for being valid
012 * command words. If the user entered an invalid command (a word that is not
013 * known) then the command word is null.
014 *
015 * If the command had only one word, then the second word is null.
016 *
017 * @author Michael Kolling and David J. Barnes + D.Bureau + C.Diouy
018 * @version 2008.03.30 + 2019.09.25 + 2022.02.11
019 */
020public class Command {
021    private CommandWord aCommandWord;
022    private String aSecondWord;
023
024    /**
025     * Create a command object. First and second word must be supplied, but
026     * either one (or both) can be null. The command word should be null to
027     * indicate that this was a command that is not recognised by this game.
028     * 
029     * @param pCommandWord first word like "go" or "help"
030     * @param pSecondWord  second word if it's needed like "north"
031     */
032    public Command(final CommandWord pCommandWord, final String pSecondWord) {
033        this.aCommandWord = pCommandWord;
034        this.aSecondWord = pSecondWord != null ? pSecondWord.toLowerCase() : null;
035    }
036
037    /**
038     * This String get the command word
039     * 
040     * @return the command word (the first word) of this command. If the
041     *         command was not understood, the result is null.
042     */
043    public CommandWord getCommandWord() {
044        return this.aCommandWord;
045    }
046
047    /**
048     * This String get the second word
049     * 
050     * @return the second word of this command. Returns null if there was no
051     *         second word.
052     */
053    public String getSecondWord() {
054        return this.aSecondWord;
055    }
056
057    /**
058     * This boolean check if there is a second word
059     * 
060     * @return true if the command has a second word.
061     */
062    public boolean hasSecondWord() {
063        return this.aSecondWord != null;
064    }
065
066    /**
067     * This boolean check if the command is known
068     * 
069     * @return true if this command was not understood.
070     */
071    public boolean isUnknown() {
072        return this.aCommandWord == CommandWord.UNKNOWN;
073    }
074} // Command

001package pkg_Command;
002import java.util.HashMap;
003import java.util.Set;
004
005/**
006 * This class is part of the "Zuul GOTY Edition" application.
007 * "Zuul GOTY Edition" is a very simple, text based adventure game.
008 * 
009 * This class holds an enumeration table of all command words known to the game.
010 * It is used to recognise commands as they are typed in.
011 *
012 * @author Michael Kolling and David J. Barnes + D.Bureau + C.Diouy
013 * @version 2008.03.30 + 2019.09.25 + 2022.02.25
014 */
015public class CommandWords {
016    // A mapping between a command word and the CommandWord
017    // associated with it.
018    private HashMap<String, CommandWord> aValidCommands;
019
020    /**
021     * Constructor - initialise the command words.
022     */
023    public CommandWords() {
024        aValidCommands = new HashMap<String, CommandWord>();
025        for (CommandWord vCommand : CommandWord.values()) {
026            if (vCommand != CommandWord.UNKNOWN) {
027                aValidCommands.put(vCommand.toString(), vCommand);
028            }
029        }
030    } // CommandWords()
031
032    /**
033     * Check whether a given String is a valid command word.
034     * 
035     * @param pString command entered by the player
036     * @return true if it is, false if it isn't.
037     */
038    public boolean isCommand(String pString) {
039        return aValidCommands.containsKey(pString);
040    }
041
042    /**
043     * Find the CommandWord associated with a command word.
044     * 
045     * @param pCommandWord The word to look up.
046     * @return The CommandWord correspondng to commandWord, or UNKNOWN
047     *         if it is not a valid command word.
048     */
049    public CommandWord getCommandWord(String pCommandWord) {
050        CommandWord vCommand = aValidCommands.get(pCommandWord);
051        if (vCommand != null) {
052            return vCommand;
053        } else {
054            return CommandWord.UNKNOWN;
055        }
056    }
057
058    /**
059     * This function get all commands in a String
060     * 
061     * @return a String of all valid commands.
062     */
063    public String getCommandList() {
064        String vCommandString = "";
065        Set<String> vKeys = aValidCommands.keySet();
066        for (String vCommand : vKeys) {
067            vCommandString += " " + vCommand;
068        }
069        return vCommandString;
070    }
071} // CommandWords

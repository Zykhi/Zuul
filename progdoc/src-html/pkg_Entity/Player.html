146    /**
147     * Try to go to one direction. If there is an exit, enter the new
148     * room, otherwise print an error message.
149     * 
150     * @param pDirection use for check if there are second word
151     */
152    public void goRoom(final Command pDirection) {
153        if (!pDirection.hasSecondWord()) {
154            // if there is no second word, we don't know where to go...
155            this.aGui.println("Go where?");
156            return;
157        }
158        this.getPreviousRooms().push(this.getCurrentRoom());
159
160        String vDirection = pDirection.getSecondWord();
161        Door vDoor = this.getCurrentRoom().getDoor(vDirection);
162
163        // Try to leave current room.
164        Room vNextRoom = this.getCurrentRoom().getExit(vDirection);
165        if (vNextRoom == null) {
166            this.aGui.println("There is no door!");
167        } else if (vDoor != null) {
168            if (vDoor.isTrap()) {
169                this.aGui.println("its a trap");
170                clearStack();
171                return;
172            }
173        } else {
174            changeRoom(vNextRoom);
175        }
176    }

178    /**
179     * This method is called when the player change the current room
180     * 
181     * @param vNextRoom the next room
182     */
183    private void changeRoom(Room vNextRoom) {
184
185        this.setRoom(vNextRoom);
186        this.aGui.println("");
187        showImage();
188        printLocationInfo();
189        if (aCurrentRoom.getCharacter() != null) {
190            showDialog();
191            // this.aGui.playDialogSound(this.aCurrentRoom.getCharacterName());
192        } else {
193            this.aGui.hideCharacterPanel();
194        }
195        this.aCurrentRoom.updateNbrRoom();
196        if (this.aGui.isSound() == false) {
197            this.aGui.soundOff();
198        } else {
199            this.aGui.playRoomSound();
200        }
201        aMovement -= 1;
202
203        if (this.getMovement() == 40) {
204            int vDelay = 100;// specify the delay for the timer
205            Timer vTimer = new Timer(vDelay, e -> {
206                // The following code will be executed once the delay is reached
207                this.printWelcome();
208            });
209            vTimer.setRepeats(false);// make sure the timer only runs once
210            vTimer.start();
211
212            int vDelay2 = 26000;// specify the delay for the timer
213            Timer vTimer2 = new Timer(vDelay2, e -> {
214                // The following code will be executed once the delay is reached
215                this.aGui.println("Type '" + CommandWord.HELP.toString() + "' if you need help." + '\n'
216                        + "You have 20 minutes to escape from the dungeon before being trapped forever" + '\n');
217                this.printLocationInfo();
218                this.aGui.enable(true);
219                this.aGui.startTimer();
220            });
221            vTimer2.setRepeats(false);// make sure the timer only runs once
222            vTimer2.start();
223        }
224    }

504        }

609    /**
610     * This method is called when enemy attack
611     *
612     * @param pMove Number of the move in the table
613     */
614    private void enemyAttackString(int pMove) {
615        String vName = this.aCurrentRoom.getCharacter().getName();
616        Entity vEnemy = this.aCurrentRoom.getCharacter();
617        if (isMissed()) {
618            this.aGui.printlnBattle(vName + " missed");
619            this.aGui.printlnBattle("");
620            setMissed(false);
621            return;
622        } else {
623            this.aGui.printlnBattle(vName + " used " + vEnemy.getAttackString(pMove));
624            this.aGui.printlnBattle("");
625        }
626    }
